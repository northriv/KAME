00102         changeFunction();

00104     catch (XKameError &e) {
00105                 e.print(getLabel() + " " + KAME::i18n("DMM Error"));
00106                 afterStop();

00109         
00110     m_lsnOnFunctionChanged = 
00111         function()->onValueChanged().connectWeak(
00112                         shared_from_this(), &XDMM::onFunctionChanged);    
00113         while(!terminated)
00114     {
00115                 msecsleep(*waitInms());
00116                 if(function()->to_str().empty()) continue;
00117       
00118                 double x;
00119                 XTime time_awared = XTime::now();
00120                 // try/catch exception of communication errors
00121                 try {
00122                         x = oneShotRead();
00123                 }
00124                 catch (XKameError &e) {
00125                         e.print(getLabel() + " " + KAME::i18n("DMM Read Error"));
00126                         continue;
00127                 }
00128                 clearRaw();
00129                 push(x);
00130                 finishWritingRaw(time_awared, XTime::now());
00131     }
00132     
00133     m_lsnOnFunctionChanged.reset();
00134         
00135     afterStop();
