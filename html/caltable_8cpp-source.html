00090 void
00091 XConCalTable::onDisplayTouched(const shared_ptr<XNode> &)
00092 {
00093   shared_ptr<XThermometer> thermo = *thermometer();
00094   if(!thermo) {
00095       m_wave->clear();
00096       return;
00097   }
00098   const int length = 1000;
00099   double step = (log(*thermo->tempMax()) - log(*thermo->tempMin())) / length;
00100   {   XScopedWriteLock<XWaveNGraph> lock(*m_wave);
00101       m_wave->setRowCount(length);
00102       double lt = log(*thermo->tempMin());
00103       for(int i = 0; i < length; i++)
00104         {
00105           double t = exp(lt);
00106           double r = thermo->getRawValue(t);
00107           m_wave->cols(0)[i] = t;
00108           m_wave->cols(1)[i] = r;
00109           m_wave->cols(2)[i] = thermo->getTemp(r) - t;
00110           lt += step;
00111         }
00112   }
00113   m_waveform->show();
00114   m_waveform->raise();  
00115 }
