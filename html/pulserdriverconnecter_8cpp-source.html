00110 XPulserDriverConnector::UpdateGraph(bool checkselection)
00111 {
00112     Graph->Graph->suspendUpdate();
00113     BarGraph->clearAllPoints();
00114     BarGraph->MaxCount->value(Pulser->RelPatList.size());
00115     for(deque<XXYPlot *>::iterator it = Plots.begin(); it != Plots.end(); it++)
00116     {
00117         (*it)->clearAllPoints();
00118         (*it)->MaxCount->value(Pulser->RelPatList.size()*2);
00119     }
00120     int i = 0;
00121     Pulser->PatternLock.readLock();
00122     unsigned long lastpat = Pulser->RelPatList.empty() ? 0 :
00123         Pulser->RelPatList[Pulser->RelPatList.size() - 1].pattern;
00124     double firsttime = -0.001, lasttime = 100;
00125 
00126     for(XPulser::tRelPatListIterator it = Pulser->RelPatList.begin(); it != Pulser->RelPatList.end(); it++)
00127     {
00128         if(Table->isRowSelected(i))
00129         {
00130             if(firsttime < 0) firsttime = it->time;
00131             lasttime = it->time;
00132         }
00133         BarGraph->addPoint(it->time, Plots.size());
00134         for(int j = 0; j < (int)Plots.size(); j++)
00135         {
00136             Plots[j]->addPoint(it->time, j + 0.7 * ((lastpat >> j) % 2));
00137             Plots[j]->addPoint(it->time, j + 0.7 * ((it->pattern >> j) % 2));
00138         }
00139         lastpat = it->pattern;       
00140         i++;
00141     }
00142     Pulser->PatternLock.readUnlock();
00143     if(checkselection)
00144     {
00145         if(lasttime == firsttime) {
00146             firsttime -= 0.5;
00147             lasttime += 0.5;
00148         }
00149         double width = lasttime - firsttime;
00150         firsttime -= width / 10;
00151         lasttime += width / 10;
00152         Graph->Graph->Axes[0].AutoScale->value(false);
00153         Graph->Graph->Axes[0].Min->value(firsttime);
00154         Graph->Graph->Axes[0].Max->value(lasttime);
00155     }        
00156     Graph->Graph->resumeUpdate();
00157 }
