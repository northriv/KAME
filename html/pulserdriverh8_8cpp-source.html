00070 int
00071 XH8Pulser::pulseAdd(uint64_t term, unsigned short pattern)
00072 {
00073         term = std::max(term, (uint64_t)lrint(MIN_PULSE_WIDTH / TIMER_PERIOD));
00074 
00075         uint32_t ulen = (uint32_t)((term - 1) / 0x8000uLL);
00076         uint32_t llen = (uint32_t)((term - 1) % 0x8000uLL);
00077 
00078         switch(ulen)
00079         {
00080                 h8ushort x;
00081         case 0:
00082                 x.msb = llen / 0x100;
00083                 x.lsb = llen % 0x100;
00084                 m_zippedPatterns.push_back(x);
00085                 x.msb = pattern / 0x100;
00086                 x.lsb = pattern % 0x100;
00087                 m_zippedPatterns.push_back(x);
00088                 break;
00089         default:
00090                 x.msb = (ulen % 0x8000u + 0x8000u) / 0x100;
00091                 x.lsb = (ulen % 0x8000u + 0x8000u) % 0x100;
00092                 m_zippedPatterns.push_back(x);
00093                 x.msb = (ulen / 0x8000u) / 0x100;
00094                 x.lsb = (ulen / 0x8000u) % 0x100;
00095                 m_zippedPatterns.push_back(x);
00096                 x.msb = (llen + 0x8000u) / 0x100;
00097                 x.lsb = (llen + 0x8000u) % 0x100;
00098                 m_zippedPatterns.push_back(x);
00099                 x.msb = pattern / 0x100;
00100                 x.lsb = pattern % 0x100;
00101                 m_zippedPatterns.push_back(x);
00102                 break;
00103         }
00104         return 0;
00105 }
