00068 int
00069 XH8Pulser::pulseAdd(uint64_t term, unsigned short pattern)
00070 {
00071         term = std::max(term, (uint64_t)lrint(MIN_PULSE_WIDTH / TIMER_PERIOD));
00072 
00073         uint32_t ulen = (uint32_t)((term - 1) / 0x8000uLL);
00074         uint32_t llen = (uint32_t)((term - 1) % 0x8000uLL);
00075 
00076         switch(ulen)
00077     {
00078                 h8ushort x;
00079     case 0:
00080                 x.msb = llen / 0x100;
00081                 x.lsb = llen % 0x100;
00082                 m_zippedPatterns.push_back(x);
00083                 x.msb = pattern / 0x100;
00084                 x.lsb = pattern % 0x100;
00085                 m_zippedPatterns.push_back(x);
00086                 break;
00087     default:
00088                 x.msb = (ulen % 0x8000u + 0x8000u) / 0x100;
00089                 x.lsb = (ulen % 0x8000u + 0x8000u) % 0x100;
00090                 m_zippedPatterns.push_back(x);
00091                 x.msb = (ulen / 0x8000u) / 0x100;
00092                 x.lsb = (ulen / 0x8000u) % 0x100;
00093                 m_zippedPatterns.push_back(x);
00094                 x.msb = (llen + 0x8000u) / 0x100;
00095                 x.lsb = (llen + 0x8000u) % 0x100;
00096                 m_zippedPatterns.push_back(x);
00097                 x.msb = pattern / 0x100;
00098                 x.lsb = pattern % 0x100;
00099                 m_zippedPatterns.push_back(x);
00100                 break;
00101     }
00102         return 0;
00103 }
