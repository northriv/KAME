00042 FIR::setupBPF(int taps, double bandwidth, double center)
00043 {
00044         if(taps < 3) taps = 2;
00045         taps = taps/2;
00046         if((m_Taps != taps) || (m_BandWidth != bandwidth) || (m_Center != center)) {
00047                 m_Taps = taps;
00048                 m_BandWidth = bandwidth;
00049                 m_Center = center;
00050                 m_Coeff.resize(taps*2 + 1);
00051                 fill(m_Coeff.begin(), m_Coeff.end(), 0.0);
00052                 m_Z.resize(taps*2 + 1);
00053                 m_Coeff[taps] = 1.0;
00054                 if(bandwidth < 1.0) {
00055                         float omega = PI*bandwidth;
00056                         float omega2 = 2 * PI * center;
00057                         for(int i = -taps; i < 0; i++) {
00058                                 float x = i*omega;
00059                                 //sinc(x) * Hamming window
00060                                 m_Coeff[taps + i] = sinf(x)/x*(0.54 + 0.46*cosf(PI*(float)i/taps));
00061                         }
00062                         //mirroring
00063                         for(int i = 1; i <= taps; i++) {
00064                                 m_Coeff[taps + i] = m_Coeff[taps - i];
00065                         }
00066                         //scaling sum into unity
00067                         float z = 0.0;
00068                         for(int i = 0; i <= 2*taps; i++) {
00069                                 z += m_Coeff[i];
00070                         }
00071                         for(int i = 0; i <= 2*taps; i++) {
00072                                 m_Coeff[i] /= z;
00073                         }
00074                         //shift center freq
00075                         for(int i = 0; i <= 2*taps; i++) {
00076                                 m_Coeff[i] *= cos(omega2 * i);
00077                         }
00078                 }
00079         }
00080         return 0;
00081 }

00083 FIR::doFIR(double *src, double *dst, int len)
00084 {
00085         fill(m_Z.begin(), m_Z.end(), 0.0);

00088         int k = 0;
00089         for(; k < taps; k++) {
00090                 oneFIR(src[(taps - k) % len], &m_Z[0], &m_Coeff[0], taps);
00091         }
00092         //wait until the center of taps
00093         for(k = 0; k < taps; k++) {
00094                 oneFIR(src[k % len], &m_Z[0], &m_Coeff[0], taps);
00095         }
00096         //do FIR
00097         for(; k < len; k++) {
00098                 *(dst++) = oneFIR(src[k % len], &m_Z[0], &m_Coeff[0], taps);
00099         }
00100         //the rest using mirror
00101         for(k = 0; k < taps; k++) {
00102                 *(dst++) = oneFIR(src[(unsigned int)(len -k - 2) % len], &m_Z[0], &m_Coeff[0], taps);
00103         }
00104         return 0;
