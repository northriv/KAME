00020 FIRMDFT::FIRMDFT(int taps, double bandwidth, double center) :
00021         m_taps(taps), m_bandWidth(bandwidth), m_centerFreq(center) {
00022         if(taps < 3) taps = 2;
00023         taps = taps/2;
00024 
00025         int taplen = 2 * taps + 1;
00026         m_tapLen = taplen;
00027         int mdctlen = std::max(32, (int)lrint(pow(2.0, ceil(log(taplen * 8) / log(2.0)))));
00028         m_mdctLen = mdctlen;
00029         m_pBufR = (double*)fftw_malloc(sizeof(double) * mdctlen);
00030         m_pBufC = (fftw_complex*)fftw_malloc(sizeof(fftw_complex) * (mdctlen / 2 + 1));
00031         m_firWnd.resize(mdctlen / 2 + 1);
00032         m_mdctWnd.resize(mdctlen - 2 * taplen);
00033         m_rdftplan = fftw_plan_dft_r2c_1d(mdctlen, m_pBufR, m_pBufC, FFTW_ESTIMATE);
00034         m_ridftplan = fftw_plan_dft_c2r_1d(mdctlen, m_pBufC, m_pBufR, FFTW_ESTIMATE);
00035         
00036         double omega = M_PI * bandwidth;
00037         for(int i = 0; i < mdctlen; i++)
00038                 m_pBufR[i] = 0.0;
00039         double z = 0.0;
00040         for(int i = -taps; i <= taps; i++) {
00041                 double x = i * omega;
00042                 //sinc(x) * Hamming window
00043                 double y = (i == 0) ? 1.0 : (sin(x)/x);
00044                 y *= 0.54 + 0.46*cos(M_PI*(double)i/taps);
00045                 m_pBufR[(mdctlen + i) % mdctlen] = y;
00046                 z += y;
00047         }
00048         //scaling sum into unity
00049         //shift center freq
00050         double omega_c = 2 * M_PI * center;
00051         for(int i = -taps; i <= taps; i++) {
00052                 m_pBufR[(mdctlen + i) % mdctlen] *= cos(omega_c * i) / (z * (double)(mdctlen));
00053         }
00054         
00055         fftw_execute(m_rdftplan);
00056         
00057         for(int i = 0; i < (int)m_firWnd.size(); i++) {
00058                 m_firWnd[i] = m_pBufC[i][0];
00059         }
00060         for(int i = 0; i < (int)m_mdctWnd.size(); i++) {
00061                 m_mdctWnd[i] = sin(M_PI * (i + 0.5) / (double)m_mdctWnd.size());
00062         }
00063 }
