00168 {
00169         m_waveformPos[num] = m_zippedPatterns.size();
00170         unsigned short word = (unsigned short)rint(pw / DMA_PERIOD);
00171         m_zippedPatterns.push_back(PATTERN_ZIPPED_COMMAND_DMA_HBURST);
00172         m_zippedPatterns.push_back((unsigned char)(word / 0x100) );
00173         m_zippedPatterns.push_back((unsigned char)(word % 0x100) );
00174         double dx = DMA_PERIOD / pw;
00175         double dp = 2*PI*freq*DMA_PERIOD;
00176         double z = pow(10.0, dB/20.0);
00177         for(int i = 0; i < word; i++) {
00178                 double w = z * func((i - word / 2.0) * dx) * 125.0;
00179                 double x = w * cos((i - word / 2.0) * dp + PI/4 + phase);
00180                 double y = w * sin((i - word / 2.0) * dp + PI/4 + phase);
00181                 x = max(min(x, 124.0), -124.0);
00182                 y = max(min(y, 124.0), -124.0);
00183                 m_zippedPatterns.push_back( (unsigned char)(char)rint(x) );     
00184                 m_zippedPatterns.push_back( (unsigned char)(char)rint(y) );     
00185         }

00280         unsigned short ulenl = (unsigned short)(ulen % 0x10000uL);      
00281         unsigned short dlen = (uint32_t)floor((msec / MTU_PERIOD) - ulen * (double)0x10000);

00283                 m_zippedPatterns.push_back(PATTERN_ZIPPED_COMMAND_WAIT_LONG_LONG);

00287                 m_zippedPatterns.push_back((unsigned char)(ulenh % 0x100) );
00288                 m_zippedPatterns.push_back((unsigned char)(ulenl / 0x100) );
00289                 m_zippedPatterns.push_back((unsigned char)(ulenl % 0x100) );
00290         }
00291         else { if(ulenl) {
00292                 m_zippedPatterns.push_back(PATTERN_ZIPPED_COMMAND_WAIT_LONG);
00293                 m_zippedPatterns.push_back((unsigned char)(dlen / 0x100) );
00294                 m_zippedPatterns.push_back((unsigned char)(dlen % 0x100) );
00295                 m_zippedPatterns.push_back((unsigned char)(ulenl / 0x100) );
00296                 m_zippedPatterns.push_back((unsigned char)(ulenl % 0x100) );

00298          else {
00299                 m_zippedPatterns.push_back(PATTERN_ZIPPED_COMMAND_WAIT);
00300                 m_zippedPatterns.push_back((unsigned char)(dlen / 0x100) );
00301                 m_zippedPatterns.push_back((unsigned char)(dlen % 0x100) );
00302           }
00303         }

00305         msec = max(0.0, msec);
00306         m_zippedPatterns.push_back(PATTERN_ZIPPED_COMMAND_DMA_SET);
