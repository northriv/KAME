00279                 m_zippedPatterns.push_back((unsigned char)(ulenl % 0x100) );
00280         }
00281         else { if(ulenl) {
00282                 m_zippedPatterns.push_back(PATTERN_ZIPPED_COMMAND_WAIT_LONG);
00283                 m_zippedPatterns.push_back((unsigned char)(dlen / 0x100) );
