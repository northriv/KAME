00034 struct Vector4 {
00035     Vector4() : x(0), y(0), z(0), w(1) {}
00036     Vector4(const Vector4<T> &v) : x(v.x), y(v.y), z(v.z), w(v.w) {}
00037     Vector4(T nx, T ny, T nz = 0, T nw = 1) : x(nx), y(ny), z(nz), w(nw) {} 
00038     T x, y, z, w;
00040     bool operator==(const Vector4<T> &s1)  const {return ((x == s1.x) && (y == s1.y) && (z == s1.z));}
00041     Vector4<T> &operator+=(const Vector4<T> &s1) {
00042         x += s1.x; y += s1.y; z += s1.z;
00043         return *this;
00044     }
00045     Vector4<T> &operator-=(const Vector4<T> &s1) {
00046         x -= s1.x; y -= s1.y; z -= s1.z;
00047         return *this;
00048     }
00049     Vector4<T> &operator*=(T k) {
00050         x *= k; y *= k; z *= k;
00051         return *this;
00052     }
00054     T distance2(const Vector4<T> &s1) const {
00055     T x1 = x - s1.x;
00056     T y1 = y - s1.y;
00057     T z1 = z - s1.z;
00058         return x1*x1 + y1*y1 + z1*z1;
00059     }
00061     T distance2(const Vector4<T> &s1, const Vector4<T> &s2) const  {
00062     T x1 = x - s1.x;
00063     T y1 = y - s1.y;
00064     T z1 = z - s1.z;
00065     T x2 = s2.x - s1.x;
00066     T y2 = s2.y - s1.y;
00067     T z2 = s2.z - s1.z;
00068     T zbab = x1*x2 + y1*y2 + z1*z2;
00069     T ab2 = x2*x2 + y2*y2 + z2*z2;
00070     T zb2 = x1*x1 + y1*y1 + z1*z1;
00071     return (zb2*ab2 - zbab*zbab) / ab2;
00072     }
00073     void normalize() {
00074         T ir = (T)1.0 / sqrtf(x*x + y*y + z*z);
00075         x *= ir; y *= ir; z *= ir;
00076     }
00077     Vector4<T> &vectorProduct(const Vector4<T> &s1) {
00078     Vector4<T> s2;
00079         s2.x = y * s1.z - z * s1.y;
00080         s2.y = z * s1.x - x * s1.z;
00081         s2.z = x * s1.y - y * s1.x;
00082         *this = s2;
00083         return *this;
00084     }
00085     T innerProduct(const Vector4<T> &s1) const {
00086         return x * s1.x + y * s1.y + z * s1.z;
00087     }
00088 };
