00296         atomic_shared_ptr<ListenerList> new_list(

00300             if(!it->lock())

00317         for(ListenerList_it it = new_list->begin(); it != new_list->end();) {
00318             if(shared_ptr<Listener> listener = it->lock()) {
00319                 // clean dead listeners and matching one.
00320                 if(!listener || (lx == listener)) {
00321                     it = new_list->erase(it);
00322                     continue;
00323                 }
00324             }
00325             it++;
00326         }
00327         if(new_list->empty()) new_list.reset();
00328         if(new_list.compareAndSwap(old_list, m_listeners)) break;
