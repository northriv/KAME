00242 XNMRT1::acuSpectrum(const std::deque<std::complex<double> > &wave,
00243      double df, double cf, double bw)
00244 {
00245   std::complex<double> x(0.0, 0.0);
00246   int cnt = 0;
00247   const int len = wave.size();
00248   const int width = lrint(bw / df);
00249   const int cpos = lrint(len / 2 + cf / df);
00250   for(int i = std::max(0, cpos - width/2); i <= std::min(len - 1, cpos + width / 2); i++) {
00251       x += wave[i];
00252       cnt++; 
00253     }
00254   return x / (double)cnt;
00255 }

00281 XNMRT1::analyze(const shared_ptr<XDriver> &emitter) throw (XRecordError&)
00282 {
00283   const double p1min = *p1Min();
00284   const double p1max = *p1Max();
00285     
00286   if((p1min <= 0) || (p1min >= p1max)) {
00287         throw XRecordError(KAME::i18n("Invalid P1Min or P1Max."), __FILE__, __LINE__);  
00288   }
00289 
00290   const int samples = *smoothSamples();
00291   if(samples <= 10) {
00292         throw XRecordError(KAME::i18n("Invalid # of Samples."), __FILE__, __LINE__);  
00293   }
00294   if(samples >= 10000) {
00295         throw XRecordError(KAME::i18n("Too many Samples."), __FILE__, __LINE__);
00296   }
00297   
00298   const bool t2mode = *t2Mode();
00299   const shared_ptr<XNMRPulseAnalyzer> _pulse1 = *pulse1();
00300     
00301   // read spectra from NMRPulseAnalyzers
00302   if( emitter != shared_from_this() ) {
00303           ASSERT( _pulse1 );
00304           ASSERT( _pulse1->time() );
00305       shared_ptr<XPulser> _pulser = *pulser();
00306       shared_ptr<XNMRPulseAnalyzer> _pulse2 = *pulse2();
00307       ASSERT( _pulser );
00308       ASSERT( _pulser->time() );
00309       ASSERT( emitter != _pulser );
00310             
00311       bool _active = *active();
00312       if(_pulse1->time() - _pulser->time() < _pulser->periodicTermRecorded() * 1e-3) {
00313             m_statusPrinter->printWarning(KAME::i18n("Too small avg. count for phase cycling."));
00314       }
00315       
00316       std::complex<double> cmp1, cmp2;
00317       cmp1 = acuSpectrum(_pulse1->ftWave(), _pulse1->dFreq() / 1000.0, *freq(), *bandWidth() );
00318       if(_pulse2) {
00319         cmp2 = acuSpectrum(_pulse2->ftWave(), _pulse2->dFreq() / 1000.0, *freq(), *bandWidth() );
00320       }    
00321       RawPt pt1, pt2;
00322       switch(_pulser->combModeRecorded()) {
00323         default:
00324            throw XRecordError(KAME::i18n("Unknown Comb Mode!"), __FILE__, __LINE__);
00325         case XPulser::N_COMB_MODE_COMB_ALT:
00326           if(t2mode) throw XRecordError(KAME::i18n("Do not use T2 mode!"), __FILE__, __LINE__);
00327           ASSERT(_pulse2);
00328             pt1.p1 = _pulser->combP1Recorded();
00329             pt1.c = (cmp1 - cmp2) / cmp1;
00330             pt1.isigma2 = 1/(_pulse1->noisePower());
00331             m_pts.push_back(pt1);
00332             break;
00333         case XPulser::N_COMB_MODE_P1_ALT:
00334           if(t2mode) 
00335                 throw XRecordError(KAME::i18n("Do not use T2 mode!"), __FILE__, __LINE__);
00336           ASSERT(_pulse2);
00337             pt1.p1 = _pulser->combP1Recorded();
00338             pt1.c = cmp1;
00339             pt1.isigma2 = 1/(_pulse1->noisePower());
00340             m_pts.push_back(pt1);
00341             pt2.p1 = _pulser->combP1AltRecorded();
00342             pt2.c = cmp2;
00343             pt2.isigma2 = 1/(_pulse2->noisePower());
00344             m_pts.push_back(pt2);
00345             break;
00346         case XPulser::N_COMB_MODE_ON:
00347           if(!t2mode) {
00348                 pt1.p1 = _pulser->combP1Recorded();
00349                 pt1.c = cmp1;
00350                 pt1.isigma2 = 1/(_pulse1->noisePower());
00351                 m_pts.push_back(pt1);
00352                 break;
00353            }
00354            m_statusPrinter->printWarning(KAME::i18n("T2 mode with comb pulse!"));
00355         case XPulser::N_COMB_MODE_OFF:
00356            if(!t2mode) {
00357                      m_statusPrinter->printWarning(KAME::i18n("Do not use T1 mode! Skipping."));
00358                      throw XSkippedRecordError(__FILE__, __LINE__);
00359            }
00360           //T2 measurement
00361             pt1.p1 = 2.0 * _pulser->tauRecorded();
00362             pt1.c = cmp1;
00363             pt1.isigma2 = 1/(_pulse1->noisePower());
00364             m_pts.push_back(pt1);
00365             break;
00366         }
00367 
00368       //set new P1s
00369       if(_active) {
00370           double x = ((double)KAME::rand())/RAND_MAX;
00371           double np1, np2;
00372           np1 = f(x);
00373           np2 = f(1-x);
00374           _pulser->output()->value(false);
00375           if(!t2mode) {
00376             _pulser->combP1()->value(np1);
00377             _pulser->combP1Alt()->value(np2);
00378           }
00379           else {
00380             _pulser->tau()->value(np1 / 2.0);
00381           }
00382           _pulser->output()->value(true);
00383         }
00384     }
00385   
00386   m_sumpts.clear();
00387   
00388   if(m_timeClearRequested > _pulse1->timeAwared()) {
00389     m_pts.clear();
00390         m_wave->clear();
00391         m_fitStatus->value("");
00392     throw XSkippedRecordError(__FILE__, __LINE__);
00393   }
00394   
00395   shared_ptr<XRelaxFunc> func = *relaxFunc();
00396   if(!func) {
00397     throw XRecordError(KAME::i18n("Please select relaxation func."), __FILE__, __LINE__);  
00398   }
00399   
00400   m_sumpts.resize(samples);
00401 
00402   struct Pt dummy;
00403   dummy.c = 0; dummy.p1 = 0; dummy.isigma = 0;
00404   std::fill(m_sumpts.begin(), m_sumpts.end(), dummy);
00405   for(std::deque<RawPt>::iterator it = m_pts.begin(); it != m_pts.end(); it++) {
00406 #define invf(x) (log((x) / p1min) / log(p1max/p1min))
00407       int idx = lrint(m_sumpts.size() * invf(it->p1));
00408       if((idx < 0) || (idx >= (int)m_sumpts.size())) continue;
00409       m_sumpts[idx].isigma += it->isigma2;
00410       m_sumpts[idx].p1 += it->isigma2 * it->p1;
00411       m_sumpts[idx].c += it->isigma2 * it->c;
00412     }
00413 
00414   double ph = *phase() / 180.0 * PI;
00415   if(*autoPhase()) {
00416       double maxcorr = -1;
00417       for(double res = 360.0/5; res > 0.01; res /= 4) {
00418         double nph = ph;
00419         for(double i = -res*2; i <= res*2; i+=res) {
00420            double tph = ph + i/180.0*PI;
00421            std::complex<double> cph(cos(tph), -sin(tph));
00422            double corr = 0.0;
00423            double sum_re = 0.0;
00424            double sum_t = 0.0;
00425            int n = 0;
00426           for(std::deque<struct Pt>::iterator it = m_sumpts.begin(); it != m_sumpts.end(); it++) {
00427             if(it->isigma == 0) continue;
00428             double re = std::real(it->c * cph);
00429             double t = log10(it->p1 / it->isigma);
00430             sum_re += re;
00431             sum_t += t;
00432             n++;
00433             corr += re * t;
00434           }
00435           if(n > 0) {
00436             corr -= sum_re*sum_t/n;
00437             corr *= (t2mode ? -1 : 1);
00438             if(corr > maxcorr) 
00439             {
00440                maxcorr = corr;
00441                nph = tph;
00442             }
00443           }
00444          }
00445          ph = nph;
00446        }
00447        ph -= 2*PI*rint(ph / 2 / PI);
00448        phase()->value(ph / PI * 180);
00449     }
00450   std::complex<double> cph(cos(ph), -sin(ph));
00451   for(std::deque<Pt>::iterator it = m_sumpts.begin(); it != m_sumpts.end(); it++) {
00452       if(it->isigma == 0) continue;
00453       it->p1 = it->p1 / it->isigma;
00454       it->c *= cph / it->isigma;
00455       it->var = (*absFit()) ? std::abs(it->c) : std::real(it->c);
00456       it->isigma = sqrt(it->isigma);
00457     }
00458 
00459   m_fitStatus->value(iterate(func, 4));
00460 
00461   t1inv()->value(1000.0 * m_params[0]);
00462   t1invErr()->value(1000.0 * m_errors[0]);
00463 }
