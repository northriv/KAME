00243                                         double df, double cf, double bw)
00244 {
00245         std::complex<double> x(0.0, 0.0);
00246         int cnt = 0;
00247         const int len = wave.size();
00248         const int width = lrint(bw / df);
00249         const int cpos = lrint(len / 2 + cf / df);
00250         for(int i = std::max(0, cpos - width/2); i <= std::min(len - 1, cpos + width / 2); i++) {
00251                 x += wave[i];
00252                 cnt++; 
00253     }
00254         return x / (double)cnt;

00441                                                 nph = tph;
00442                                         }
00443                                 }
00444                         }

00446                 }
00447                 ph -= 2*PI*rint(ph / 2 / PI);
00448                 phase()->value(ph / PI * 180);
00449     }
00450         std::complex<double> cph(cos(ph), -sin(ph));
00451         for(std::deque<Pt>::iterator it = m_sumpts.begin(); it != m_sumpts.end(); it++) {
00452                 if(it->isigma == 0) continue;
00453                 it->p1 = it->p1 / it->isigma;

00456                 it->isigma = sqrt(it->isigma);
00457     }
