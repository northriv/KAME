00282         const std::vector< std::complex<double> >&wave, int origin, double cf,
00283         std::deque<std::complex<double> > &value_by_cond) {
00284         value_by_cond.clear();
00285         std::deque<FFT::twindowfunc> funcs;
00286         m_solver->windowFuncs(funcs);
00287 
00288         int idx = 0;
00289         for(std::deque<double>::iterator wit = m_windowWidthList.begin(); wit != m_windowWidthList.end(); wit++) {
00290                 for(std::deque<FFT::twindowfunc>::iterator fit = funcs.begin(); fit != funcs.end(); fit++) {
00291                         if(m_convolutionCache.size() <= idx) {
00292                                 m_convolutionCache.push_back(shared_ptr<ConvolutionCache>(new ConvolutionCache));
00293                         }
00294                         shared_ptr<ConvolutionCache> cache = m_convolutionCache[idx];
00295                         if((cache->windowwidth != *wit) || (cache->origin != origin) ||
00296                                 (cache->windowfunc != *fit) || (cache->cfreq != cf) ||
00297                                 (cache->wave.size() != wave.size())) {
00298                                 cache->windowwidth = *wit;
00299                                 cache->origin = origin;
00300                                 cache->windowfunc = *fit;
00301                                 cache->cfreq = cf;
00302                                 cache->power = 0.0;
00303                                 cache->wave.resize(wave.size());
00304                                 double wk = 1.0 / FFTSolver::windowLength(wave.size(), -origin, *wit);
00305                                 for(int i = 0; i < (int)wave.size(); i++) {
00306                                         double w = (*fit)((i - origin) * wk) / (double)wave.size();
00307                                         cache->wave[i] = std::polar(w, -2.0*M_PI*cf*(i - origin));
00308                                         cache->power += w*w;
00309                                 }
00310                         }
00311                         
00312                         std::complex<double> z(0.0);
00313                         for(int i = 0; i < (int)cache->wave.size(); i++) {
00314                                 z += wave[i] * cache->wave[i];
00315                         }
