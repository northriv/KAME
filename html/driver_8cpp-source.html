00083 XRecordDependency::merge(const shared_ptr<const XDriver> &driver)
00084 {
00085     if(m_bConflict) return true;
00086     bool conflicted = false;
00087     if(!driver->time()) {
00088         conflicted = true;
00089     }
00090     else {
00091         m_dependency.insert(tDependency(driver, driver->time()));
00092         shared_ptr<XRecordDependency> dep = driver->dependency();
00093         if(dep) {
00094             if(dep->isConflict()) {
00095                 conflicted = true;
00096             }
00097             else {
00098                 for(tDependency_it dit = dep->m_dependency.begin(); 
00099                     dit != dep->m_dependency.end(); dit++) {
00101                     tDependency_it fit = m_dependency.find(*dit);
00102                     if(fit != m_dependency.end()) {
00103                         ASSERT(fit->driver == dit->driver);
00104                         if(fit->time != dit->time) {
00106                             conflicted = true;
00107                             break;
00108                         }
00109                     }
00110                     else {
00111                         m_dependency.insert(*dit);
00112                     }
00113                 }
00114             }
00115         }
00116     }
00117     m_bConflict = conflicted;
00118     return conflicted;
00119 }
