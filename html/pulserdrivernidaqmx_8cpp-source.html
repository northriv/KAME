00388 template <typename T>
00389 inline T *
00390 fastFill(T* p, T x, unsigned int cnt) {
00391         if(cnt > 100) {
00392                 for(;(int)p % (sizeof(uint64_t) / sizeof(T)); cnt--)
00393                         *p++ = x;
00394                 uint64_t *pp = (uint64_t *)p;
00395                 union {
00396                         uint64_t dw; T w[sizeof(uint64_t) / sizeof(T)];
00397                 } pack;
00398                 for(unsigned int i = 0; i < (sizeof(uint64_t) / sizeof(T)); i++)
00399                         pack.w[i] = x;
00400                 unsigned int pcnt = cnt / (sizeof(uint64_t) / sizeof(T));
00401                 cnt = cnt % (sizeof(uint64_t) / sizeof(T));
00402                 for(unsigned int i = 0; i < pcnt; i++)
00403                         *pp++ = pack.dw;
00404                 p = (T*)pp;
00405         }
00406         for(unsigned int i = 0; i < cnt; i++)
00407                 *p++ = x;
00408         return p;
00409 }
