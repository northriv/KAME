00170 XNIDAQmxPulser::makeWaveForm(int num, double pw, tpulsefunc func, double dB, double freq, double phase)
00171 {
00172         m_waveformPos[num] = m_zippedPatterns.size();
00173         unsigned short word = (unsigned short)rint(pw / DMA_PERIOD);
00174         m_zippedPatterns.push_back(PATTERN_ZIPPED_COMMAND_DMA_HBURST);
00175         m_zippedPatterns.push_back((unsigned char)(word / 0x100) );
00176         m_zippedPatterns.push_back((unsigned char)(word % 0x100) );
00177         double dx = DMA_PERIOD / pw;
00178         double dp = 2*PI*freq*DMA_PERIOD;
00179         double z = pow(10.0, dB/20.0);
00180         for(int i = 0; i < word; i++) {
00181                 double w = z * func((i - word / 2.0) * dx) * 125.0;
00182                 double x = w * cos((i - word / 2.0) * dp + PI/4 + phase);
00183                 double y = w * sin((i - word / 2.0) * dp + PI/4 + phase);
00184                 x = max(min(x, 124.0), -124.0);
00185                 y = max(min(y, 124.0), -124.0);
00186                 m_zippedPatterns.push_back( (unsigned char)(char)rint(x) );     
00187                 m_zippedPatterns.push_back( (unsigned char)(char)rint(y) );     
00188         }

00292                 m_zippedPatterns.push_back((unsigned char)(ulenl % 0x100) );
