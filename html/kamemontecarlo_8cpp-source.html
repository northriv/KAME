00251 XMonteCarloDriver::analyzeRaw() throw (XRecordError&)
00252 {
00253     int size = MonteCarlo::length();
00254     if(rawData().size() != size*size*size*16)
00255         throw XRecordError("Size Mismatch", __FILE__, __LINE__);
00256     MonteCarlo::Vector3<double> field_dir(*m_hdirx,*m_hdiry,*m_hdirz);
00257     field_dir.normalize();
00258     MonteCarlo::Vector3<double> field(field_dir);
00259     field *= *m_targetField;
00260     char *spins = (char*)&rawData()[0];
00261     m_store->read(spins, *m_targetTemp, field);
00262 
00263     double dT = *m_targetTemp - m_lastTemp;
00264     m_lastTemp = *m_targetTemp;
00265     double m = m_store->magnetization().innerProduct(field_dir);
00266     double dM = m - m_lastMagnetization;
00267     double dH = *m_targetField - m_lastField;
00268     double dU = m_dU;
00269     // DS += (DU + HDM + MDH)/T
00270     m_sumDS += (dU + dM * *m_targetField * MU_B * N_A) / *m_targetTemp; 
00271     // DU += -MDH.
00272     dU += -m_lastMagnetization * MU_B * N_A * dH;
00273     double dUav = m_DUav * N_A;
00274     dUav += -m_Mav * MU_B * N_A * dH;
00275     double c = (fabs(dT) < 1e-10) ? 0.0 : ((m_sumDU + dUav - m_sumDUav) / dT);
00276     m_sumDUav = m_sumDU + dUav;
00277     m_sumDU += dU;
00278     m_lastMagnetization = m;
00279     m_lastField = *m_targetField;
00280     double u = m_sumDU;
00281     if(rand() % 20 == 0) {
00282         std::cerr << "Recalculate Internal Energy." << std::endl;
00283         u = m_store->internalEnergy() * N_A;
00284         if(fabs((u - m_sumDU)/u) > 1e-5) {
00285             gErrPrint(formatString("SumDU Error = %g!", (double)(u - m_sumDU)/u)); 
00286         }
00287     }
00288     MonteCarlo::Quartet quartet = m_store->siteMagnetization();
00289     m_entryT->value(m_lastTemp);
00290     m_entryH->value(m_lastField);
00291     m_entryU->value(u);
00292     m_entryC->value(c);
00293     m_entryCoT->value(c / m_lastTemp);
00294     m_entryS->value(m_sumDS);
00295     m_entryM->value(m_Mav);
00296     m_entry2in2->value(100.0*quartet.twotwo);
00297     m_entry1in3->value(100.0*quartet.onethree);
00298 }
