00236 XMonteCarloDriver::analyzeRaw() throw (XRecordError&)
00237 {
00238     int size = MonteCarlo::length();
00239     if(rawData().size() != size*size*size*16)
00240         throw XRecordError("Size Mismatch", __FILE__, __LINE__);
00241     MonteCarlo::Vector3<double> field_dir(*m_hdirx,*m_hdiry,*m_hdirz);
00242     field_dir.normalize();
00243     MonteCarlo::Vector3<double> field(field_dir);
00244     field *= *m_targetField;
00245     char *spins = (char*)&rawData()[0];
00246     m_store->read(spins, *m_targetTemp, field);
00247 
00248     double dT = *m_targetTemp - m_lastTemp;
00249     m_lastTemp = *m_targetTemp;
00250     double m = m_store->magnetization().innerProduct(field_dir);
00251     double dM = m - m_lastMagnetization;
00252     double dH = *m_targetField - m_lastField;
00253     double dU = m_dU;
00254     // DS += (DU + HDM + MDH)/T
00255     m_sumDS += (dU + dM * *m_targetField * MU_B * N_A) / *m_targetTemp; 
00256     // DU += -MDH.
00257     dU += -m_lastMagnetization * MU_B * N_A * dH;
00258     double dUav = m_DUav * N_A;
00259     dUav += -m_Mav * MU_B * N_A * dH;
00260     double c = (fabs(dT) < 1e-10) ? 0.0 : ((m_sumDU + dUav - m_sumDUav) / dT);
00261     m_sumDUav = m_sumDU + dUav;
00262     m_sumDU += dU;
00263     m_lastMagnetization = m;
00264     m_lastField = *m_targetField;
00265     double u = m_sumDU;
00266     if(rand() % 20 == 0) {
00267         std::cerr << "Recalculate Internal Energy." << std::endl;
00268         u = m_store->internalEnergy() * N_A;
00269         if(fabs((u - m_sumDU)/u) > 1e-5) {
00270             gErrPrint(formatString("SumDU Error = %g!", (double)(u - m_sumDU)/u)); 
00271         }
00272     }
00273     MonteCarlo::Quartet quartet = m_store->siteMagnetization();
00274     m_entryT->value(m_lastTemp);
00275     m_entryH->value(m_lastField);
00276     m_entryU->value(u);
00277     m_entryC->value(c);
00278     m_entryCoT->value(c / m_lastTemp);
00279     m_entryS->value(m_sumDS);
00280     m_entryM->value(m_Mav);
00281     m_entry2in2->value(100.0*quartet.twotwo);
00282     m_entry1in3->value(100.0*quartet.onethree);
