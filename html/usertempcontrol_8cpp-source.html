00227 {
00228         double x;
00229         { XScopedLock<XInterface> lock(*interface());
00230         int wait = interface()->gpibWaitBeforeRead();
00231         interface()->setGPIBWaitBeforeRead(300);
00232         interface()->query("AVE 1;*OPC?");
00233         interface()->setGPIBWaitBeforeRead(wait);
00234         x = read("AVE");
00235         }      
00236         if(m_autorange_wait++ > 10)
00237         {
00238                 int range = getRange();
00239                 if(lrint(read("OVL")) == 0)
00240         {
00241                         if(fabs(x) < 0.1 * pow(10.0, range - 1))
00242                                 setRange(std::max(range - 1, 1));
00243                         if(fabs(x) > 1.6 * pow(10.0, range - 1))
00244                                 setRange(std::min(range + 1, 7));
00245         }
00246                 else
00247         {
00248                         setRange(std::min(range + 1, 7));
00249         }  
00250         }
00251         return x;
