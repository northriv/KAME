00290 void
00291 MEMBurg::step(const shared_ptr<MEMBurgContext> &context) {
00292         unsigned int t = context->t;
00293         unsigned int p = context->p;
00294         std::complex<double> x = 0.0, y = 0.0;
00295         for(unsigned int i = p + 1; i < t; i++) {
00296                 x += context->epsilon[i] * std::conj(context->eta[i-1]);
00297                 y += std::norm(context->epsilon[i]) + std::norm(context->eta[i-1]);
00298         }
00299         std::complex<double> alpha = x / y;
00300         alpha *= -2;
00301         for(unsigned int i = t - 1; i >= p + 1; i--) {
00302                 context->eta[i] = context->eta[i-1] + std::conj(alpha) * context->epsilon[i];
00303                 context->epsilon[i] += alpha * context->eta[i-1];
00304         }
00305         std::vector<std::complex<double> > a_next(p + 2);
00306         for(unsigned int i = 0; i < p + 2; i++) {
00307                 a_next[i] = context->a[i] + alpha * std::conj(context->a[p + 1 - i]);
00308         }
00309         std::copy(a_next.begin(), a_next.end(), context->a.begin());
00310         context->sigma2 *= 1 - std::norm(alpha);
00311 }

00410                 for(it = 0; it < 50; it++) {
00411                         err = stepMEM(memin, memout, alpha, sigma, t0, torr);
