00051     if(vnode) {
00052         if(list) {
00053             for(int j = 0; j < level; j++) m_ofs << "\t";
00054             if(ghost)
00055                 m_ofs << "# ";

00057         }
00058         else {
00059             if(ghost)
00060                 m_ofs << "\t# ";
00061         }
00062         QString s(vnode->to_str());
00063         s.replace( QChar('\\'), "\\\\");
00064         s.replace( QChar('\n'), "\\n");
00065         s.replace( QChar('\r'), "\\r");

00068             << (const char *)s.utf8()
00069             << "\")" 
00070             << std::endl;
00071     }
00072     else
00073         if(!list) {m_ofs << std::endl;}
00074         
00075     shared_ptr<XListNodeBase> lnode = dynamic_pointer_cast<XListNodeBase>(node);
00076     bool write_typename = false;
00077     if(lnode) {

00079         write_typename = (lnode->getTypename().find("XListNode") != 0);
00080         // XAliasListNode doesn't want creation of child
00081         if(lnode->getTypename().find("XAliasListNode") == 0) lnode.reset();

00087             for(int j = 0; j < level; j++) m_ofs << "\t";
00088             if(ghost)
00089                 m_ofs << "# ";

00095                 m_ofs << "x.last.create(";
00096                 if(write_typename || child->getName().length()) {
00097                     m_ofs << "\""
00098                         << (write_typename ? child->getTypename().c_str() : "")
00099                         << "\"";
00100                 }
00101                 if(child->getName().length()) {
00102                     m_ofs << ",\""
00103                         << child->getName()
00104                         << "\"";
00105                 }
00106                 m_ofs <<  ")";
00107             }
00108             else {
00109                 if(child->getName().length()) {
00110                     m_ofs << "x.last[\""
00111                         << child->getName()
00112                         <<  "\"]";
00113                 }
00114                 else {
00115                     m_ofs << "x.last["
00116                         << idx
00117                         <<  "]";
00118                 }
00119             }
00120             if(child_list) {
00121                 m_ofs << std::endl;

00124             if(child_list) {
