00129     shared_ptr<XPulser> pulser(m_pulser);

00132     std::deque<XGraph::ValPoint> & barplot_points(m_barPlot->points());

00139         (*it)->maxCount()->value(pulser->m_relPatList.size() * 2);

00143     uint32_t lastpat = pulser->m_relPatList.empty() ? 0 :
00144         pulser->m_relPatList[pulser->m_relPatList.size() - 1].pattern;
00145     double firsttime = -0.001, lasttime = 100;
00146     
00147     int i = 0;
00148     for(XPulser::RelPatListIterator it = pulser->m_relPatList.begin(); 
00149             it != pulser->m_relPatList.end(); it++)
00150     {
00151         double time = it->time * pulser->resolution();
00152         if(m_pTable->isRowSelected(i))
00153         {
00154             if(firsttime < 0) firsttime = time;
00155             lasttime = time;
00156         }
00157         barplot_points.push_back(XGraph::ValPoint(time, m_plots.size()));
00158         for(int j = 0; j < (int)plots_points.size(); j++)
00159         {
00160             plots_points[j]->push_back(XGraph::ValPoint(time, j + 0.7 * ((lastpat >> j) % 2)));
00161             plots_points[j]->push_back(XGraph::ValPoint(time, j + 0.7 * ((it->pattern >> j) % 2)));
00162         }
00163         lastpat = it->pattern;
00164         i++;
00165     }
00166     if(checkselection)
00167     {
00168         if(lasttime == firsttime) {
00169             firsttime -= 0.5;
00170             lasttime += 0.5;
00171         }
00172         double width = lasttime - firsttime;
00173         firsttime -= width / 10;
00174         lasttime += width / 10;
00175         shared_ptr<XAxis> axisx = *m_barPlot->axisX();
00176         axisx->autoScale()->value(false);
00177         axisx->minValue()->value(firsttime);
00178         axisx->maxValue()->value(lasttime);
00179     }
