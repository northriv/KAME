00302     Vector3<double> m;

00313 inline double
00314 MonteCarlo::flippingProbability(int site, int lidx, double h, double *pdu)
00315 {
00316     int sidx = spins_real_index(lidx);
00317     double du = 2 * readSpin(site, sidx) * A_MOMENT * MU_B * (
00318         h + m_ext_field[site]);
00319     *pdu = du;
00320 
00321     if(du <= 0.0) return 1.0;
00322 
00323     // probability.
00324     return exp(-m_beta*du);

00549                     }

00551                 }

00604 MonteCarlo::modifyReciprocalImage(Spin diff, int site1, int i, int j, int k)
00605 {
00606     int cutoff = s_cutoff_rec;
00607     int cnt = 2*cutoff + 1;
00608     complex<Spin> *pspin = &m_spins_rec[site1][0];
00609     
00610     Vector3<double> pos1(cg_ASitePositions[site1]);
00611     pos1 *= LATTICE_CONST / 4.0;
00612     double phx = -2*M_PI / (LATTICE_CONST * s_L) * (i * LATTICE_CONST + pos1.x);
00613     double phy = -2*M_PI / (LATTICE_CONST * s_L) * (j * LATTICE_CONST + pos1.y);
00614     double phz = -2*M_PI / (LATTICE_CONST * s_L) * (k * LATTICE_CONST + pos1.z);
00615     complex<Spin> exp_i_rx = exp(complex<Spin>(0.0, phx));
00616     complex<Spin> exp_i_ry = exp(complex<Spin>(0.0, phy));
00617     complex<Spin> exp_i_rz = exp(complex<Spin>(0.0, phz));
00618         
00619     complex<Spin> exp_ikrz = ((Spin)diff)
00620             * exp(complex<Spin>(0.0, -cutoff * (phx + phy)));
00621     for(int kz = 0; kz <= cutoff; kz++) {
00622         complex<Spin> exp_ikryz = exp_ikrz;
00623         for(int ky = -cutoff; ky <= cutoff; ky++) {
00624             complex<Spin> exp_ikr = exp_ikryz;
00625             for(int n = 0; n < cnt; n++) {
00626                 pspin[n] += exp_ikr;
00627                 exp_ikr *= exp_i_rx;
00628             }
00629             pspin+=cnt;
00630             exp_ikryz *= exp_i_ry;
00631         }
00632         exp_ikrz *= exp_i_rz;
00633     }
00634     ASSERT(pspin == &*m_spins_rec[site1].end());
