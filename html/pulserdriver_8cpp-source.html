00041 double bessel_i0(double x) {
00042 float y;
00043 float pow = 1.0;
00044 int s = 1;
00045 float z = (x/2)*(x/2);
00046     y = 1.0;
00047     for(int k = 1; k < 5; k++) {
00048         s *= k;
00049         pow *= z;
00050         y += pow / (s*s);
00051     }
00052     return y;
00053 }

00068 double XPulser::pulseFuncBlackmanHarris(double x) {
00069         return 0.35875+0.48829*cos(2*PI*x)+0.14128*cos(4*PI*x)+0.01168*cos(6*PI*x);
00070 }
00071 double XPulser::pulseFuncFlatTop(double x) {
00072         return pulseFuncHamming(x)*((fabs(x) < 1e-4) ? 1 : sin(4*PI*x)/(4*PI*x));
00073 }
00074 double XPulser::pulseFuncFlatTopLong(double x) {
00075         return pulseFuncHamming(x)*((fabs(x) < 1e-4) ? 1 : sin(6*PI*x)/(6*PI*x));
00076 }
00077 double XPulser::pulseFuncFlatTopLongLong(double x) {
00078         return pulseFuncHamming(x)*((fabs(x) < 1e-4) ? 1 : sin(8*PI*x)/(8*PI*x));

00080 double XPulser::pulseFuncKaiser(double x, double alpha) {
00081         x = 2*x;
00082         x = sqrt(std::max(1 - x*x, 0.0));       
00083         return bessel_i0(PI*alpha*x) / bessel_i0(PI*alpha);
00084 }
