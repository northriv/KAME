00060 double bessel_i0(double x) {
00061 float y;
00062 float pow = 1.0;
00063 int s = 1;
00064 float z = (x/2)*(x/2);
00065     y = 1.0;
00066     for(int k = 1; k < 5; k++) {
00067         s *= k;
00068         pow *= z;
00069         y += pow / (s*s);
00070     }
00071     return y;
00072 }

00078 double XPulser::pulseFuncHanning(double x) {
00079         return 0.5 + 0.5*cos(2*PI*x);
00080 }
00081 double XPulser::pulseFuncHamming(double x) {
00082         return 0.54 + 0.46*cos(2*PI*x);
00083 }
00084 double XPulser::pulseFuncBlackman(double x) {
00085         return 0.42323+0.49755*cos(2*PI*x)+0.07922*cos(4*PI*x);
00086 }
00087 double XPulser::pulseFuncBlackmanHarris(double x) {
00088         return 0.35875+0.48829*cos(2*PI*x)+0.14128*cos(4*PI*x)+0.01168*cos(6*PI*x);
00089 }
00090 double XPulser::pulseFuncFlatTop(double x) {
00091         return pulseFuncHamming(x)*((fabs(x) < 1e-4) ? 1 : sin(4*PI*x)/(4*PI*x));
00092 }
00093 double XPulser::pulseFuncFlatTopLong(double x) {
00094         return pulseFuncHamming(x)*((fabs(x) < 1e-4) ? 1 : sin(6*PI*x)/(6*PI*x));
00095 }
00096 double XPulser::pulseFuncFlatTopLongLong(double x) {
00097         return pulseFuncHamming(x)*((fabs(x) < 1e-4) ? 1 : sin(8*PI*x)/(8*PI*x));
00098 }
00099 double XPulser::pulseFuncKaiser(double x, double alpha) {
00100         x = 2*x;
00101         x = sqrt(std::max(1 - x*x, 0.0));       
00102         return bessel_i0(PI*alpha*x) / bessel_i0(PI*alpha);
00103 }
00104 double XPulser::pulseFuncKaiser1(double x) {
00105         return pulseFuncKaiser(x, 3.0);
00106 }
00107 double XPulser::pulseFuncKaiser2(double x) {
00108         return pulseFuncKaiser(x, 7.2);
00109 }
00110 double XPulser::pulseFuncKaiser3(double x) {
00111         return pulseFuncKaiser(x, 15.0);
00112 }
00113 double XPulser::pulseFuncHalfCos(double x) {
00114     return cos(PI*x);
00115 }
00116 double XPulser::pulseFuncChoppedHalfCos(double x) {
00117     return 2.0*std::min(0.5, cos(PI*x));
