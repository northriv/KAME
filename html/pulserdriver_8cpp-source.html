00039 double bessel_i0(double x) {
00040 float y;
00041 float pow = 1.0;
00042 int s = 1;
00043 float z = (x/2)*(x/2);
00044     y = 1.0;
00045     for(int k = 1; k < 5; k++) {
00046         s *= k;
00047         pow *= z;
00048         y += pow / (s*s);
00049     }
00050     return y;
00051 }

00067         return 0.35875+0.48829*cos(2*PI*x)+0.14128*cos(4*PI*x)+0.01168*cos(6*PI*x);

00069 double XPulser::pulseFuncFlatTop(double x) {
00070         return pulseFuncHamming(x)*((fabs(x) < 1e-4) ? 1 : sin(4*PI*x)/(4*PI*x));
00071 }
00072 double XPulser::pulseFuncFlatTopLong(double x) {
00073         return pulseFuncHamming(x)*((fabs(x) < 1e-4) ? 1 : sin(6*PI*x)/(6*PI*x));
00074 }
00075 double XPulser::pulseFuncFlatTopLongLong(double x) {
00076         return pulseFuncHamming(x)*((fabs(x) < 1e-4) ? 1 : sin(8*PI*x)/(8*PI*x));

00078 double XPulser::pulseFuncKaiser(double x, double alpha) {
00079         x = 2*x;
00080         x = sqrt(std::max(1 - x*x, 0.0));       
00081         return bessel_i0(PI*alpha*x) / bessel_i0(PI*alpha);
00082 }
