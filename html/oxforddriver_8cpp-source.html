00046 XOxfordInterface::query(const char *str) throw (XInterface::XCommError &)
00047 {
00048   lock();
00049   try {
00050       for(int i = 0; i < 30; i++)
00051         {
00052           XCharInterface::send(str);
00053           XCharInterface::receive();
00054           if(buffer().size() >= 1)
00055           if(buffer()[0] == str[0]) {
00056               unlock();
00057               return;
00058           }
00059           msecsleep(100);
00060         }
00061   }
00062   catch (XCommError &e) {
00063        unlock();
00064        throw e;
00065   }
00066   unlock();
00067   throw XCommError(KAME::i18n("Oxford Query Error, Initial doesn't match"), __FILE__, __LINE__);
00068 }
