00057 void
00058 XQGraphPainter::posOffAxis(const XGraph::ScrPoint &dir, 
00059     XGraph::ScrPoint *src, XGraph::SFloat offset) {
00060         XGraph::ScrPoint scr, dir_proj;
00061         windowToScreen(m_pItem->width() / 2, m_pItem->height() / 2, 0.0, &scr);
00062         windowToScreen(m_pItem->width() / 2, m_pItem->height() / 2, 1.0, &dir_proj);
00063         dir_proj -= scr;
00064         dir_proj.normalize();
00065 double x, y, z;
00066         if(screenToWindow(*src, &x, &y, &z)) return;
00067         x = x / m_pItem->width() - 0.5;
00068         y = y / m_pItem->height() - 0.5;
00069 double r1 = x*x + y*y;
00070         XGraph::ScrPoint s1 = dir_proj;
00071         s1.vectorProduct(dir);
00072         s1.normalize();
00073         s1 *= offset;
00074         *src += s1;
00075         int ret = screenToWindow(*src, &x, &y, &z);
00076         x = x / m_pItem->width() - 0.5;
00077         y = y / m_pItem->height() - 0.5;
00078 double r2 = x*x + y*y;
00079         if(ret || ((r2 - r1) * offset < 0) ) {
00080                 *src -= s1;
00081                 *src -= s1;
00082         }
00083 }

00339         }
