00373       if(!status) break;
00374       if(XTime::now() - firsttime < 0.01) continue;

00376       double p1max = *p1Max();
00377       double p1min = *p1Min();
00378       m_params[0] = 1.0 / exp(log(p1max/p1min) * (((double)KAME::rand())/RAND_MAX) + log(p1min));
00379       m_params[1] = 0.1 + 0.5*(((double)KAME::rand())/RAND_MAX);
00380       m_params[2] = 0.0;
00381       status = do_nlls(n, p, m_params, m_errors, &norm,
00382                        &nlls, &XRelaxFunc::relax_f, &XRelaxFunc::relax_df, &XRelaxFunc::relax_fdf, itercnt);
00383     }
00384   m_errors[0] *= norm / sqrt((double)n);
00385   m_errors[1] *= norm / sqrt((double)n);
00386   m_errors[2] *= norm / sqrt((double)n);
00387 
00388   double t1 = 0.001 / m_params[0];
00389   double t1err = 0.001 / pow(m_params[0], 2.0) * m_errors[0];
00390   QString buf = "";
00391   if(*t2Mode())
00392     {
00393       buf += QString().sprintf("1/T2[1/ms] = %.5f +/- %.5f (%.2f%%)\n",
00394              1000.0 * m_params[0], 1000.0 * m_errors[0], fabs(100.0 * m_errors[0]/m_params[0]));
00395       buf += QString().sprintf("T2[ms] = %.6f +/- %.6f (%.2f%%)\n",
00396              t1, t1err, fabs(100.0 * t1err/t1));
00397     }
00398   else
00399     {
00400       buf += QString().sprintf("1/T1[1/s] = %.5f +/- %.5f (%.2f%%)\n",
00401              1000.0 * m_params[0], 1000.0 * m_errors[0], fabs(100.0 * m_errors[0]/m_params[0]));
00402       buf += QString().sprintf("T1[s] = %.7f +/- %.7f (%.2f%%)\n",
00403              t1, t1err, fabs(100.0 * t1err/t1));
00404     }
00405   buf += QString().sprintf("c[V] = %.5g +/- %.5g (%.3f%%)\n",

00410   buf += QString().sprintf("rms of residuals = %.3g\n", norm / sqrt((double)n));
