00020 FFT::fitLength(int length0) {

00022         length = std::min(length, (int)lrint(pow(2.0, (ceil(log(length0 / 3.0) / log(2.0))))) * 3);

00025         length = std::min(length, (int)lrint(pow(2.0, (ceil(log(length0 / 9.0) / log(2.0))))) * 9);

00046                 std::vector<std::complex<double> >& waveout) {
00047         int size = wavein.size();

00050         const std::complex<double> *pin = &wavein[0];
00051         fftw_complex *pout = m_pBufin;
00052         for(int i = 0; i < size; i++) {
00053                 (*pout)[0] = pin->real();
00054                 (*pout)[1] = pin->imag();
00055                 pout++;
00056                 pin++;
00057         }
00058         fftw_execute(*m_fftplan);
00059         const fftw_complex *pin2 = m_pBufout;
00060         std::complex<double> *pout2 = &waveout[0];
00061         for(int i = 0; i < size; i++) {
00062                 *pout2 = std::complex<double>((*pin2)[0], (*pin2)[1]);
00063                 pout2++;
00064                 pin2++;
00065         }
