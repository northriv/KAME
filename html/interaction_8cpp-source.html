00043 int
00044 MonteCarlo::dipoleFieldReal(VectorInt v, int site2, Vector3<double> *ret)
00045 {
00046     int r2int = v.x*v.x + v.y*v.y + v.z*v.z;
00047     // spherical boundary
00048     if(r2int - 0.01 > (4*s_cutoff_real_radius)*(4*s_cutoff_real_radius)) {
00049         return false;
00050     }
00051     double r = LATTICE_CONST/4.0 * sqrt((double)r2int);
00052     double ir = 1.0/r;
00053     double alphar = s_alpha*r;
00054     double ir_eff = _erfc(alphar)*ir;
00055     double derfc = 2.0*s_alpha/sqrt(M_PI)*exp(-alphar*alphar);
00056     double ir3_eff = ir*ir*(ir_eff + derfc);
00057     double ir5_eff = ir*ir*(ir3_eff + 2.0/3.0*s_alpha*s_alpha*derfc);
00058     Vector3<double> rij((double)v.x, (double)v.y, (double)v.z);
00059     rij *= LATTICE_CONST/4.0;
00060     Vector3<double> m2(s_ASiteIsingVector[site2]);
00061     m2 *= 1e-7 * A_MOMENT * MU_B;
00062     double mr = m2.innerProduct(rij);
00063     Vector3<double> hdip(rij);
00064     hdip *= mr * ir5_eff * 3.0;
00065     m2 *= ir3_eff;
00066     hdip -= m2;
00067     *ret = hdip;
00068     return true;
00069 }
00070 int
00071 MonteCarlo::dipoleFieldRec(VectorInt v, int site2, Vector3<double> *ret)
00072 {
00073     int k2int = v.x*v.x + v.y*v.y + v.z*v.z;
00074     int ridx = reciprocal_index(v.x,v.y,v.z);
00075     if(k2int >= s_cutoff_rec_radius*s_cutoff_rec_radius) {
00076         return false;
00077     }
00078 
00079     Vector3<double> m2(s_ASiteIsingVector[site2]);
00080     m2 *= 4.0 * M_PI * 1e-7 * A_MOMENT * MU_B / LATTICE_VOLUME / (s_num_spins/16);
00081     Vector3<double> k(v.x, v.y, v.z);
00082     k *= 2.0*M_PI / (LATTICE_CONST * s_L);
00083     Vector3<double> hdip(k);
00084     if(k2int == 0) {
00085         // surface term.
00086         hdip = m2;
00087         hdip *= - s_dfactor;
00088     }
00089     else {
00090         double k2 = k.innerProduct(k);
00091         hdip *= - exp(-k2/(4.0*s_alpha*s_alpha)) / k2 * k.innerProduct(m2);
00092     }
00093     // summation of minus-k space.
00094     if(v.z != 0)
00095         hdip *= 2.0;
00096     *ret = hdip;
00097     return true;
00098 }
