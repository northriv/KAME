00258 XTempControl::pid(XTime time, double temp)
00259 {
00260         double p = *prop();
00261         double i = *interval();
00262         double d = *deriv();
00263 
00264         double dt = temp - *targetTemp();
00265         double dxdt = 0.0;
00266         double acc = 0.0;
00267         if((i > 0) && (time - m_pidLastTime < i)) {
00268                 m_pidAccum += (time - m_pidLastTime) * dt;
00269                 dxdt = (temp - m_pidLastTemp) / (time - m_pidLastTime); 
00270                 acc = m_pidAccum / i;
00271                 acc = - std::min(std::max(- acc * p, -2.0), 100.0) / p;
00272                 m_pidAccum = acc * i;
00273         }
00274         else
00275                 m_pidAccum = 0;
00276 
00277         m_pidLastTime = time;
00278         m_pidLastTemp = temp;
00279 
00280         return -(dt + acc + dxdt * d) * p;
00281 }
