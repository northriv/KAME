00250                 false, shared_from_this(), &XTempControl::onManualPowerChanged);

00252                 false, shared_from_this(), &XTempControl::onHeaterModeChanged);
00253   m_lsnOnPowerRangeChanged = powerRange()->onValueChanged().connectWeak(
00254                 false, shared_from_this(), &XTempControl::onPowerRangeChanged);
00255   m_lsnOnCurrentChannelChanged = m_currentChannel->onValueChanged().connectWeak(
00256                 false, shared_from_this(), &XTempControl::onCurrentChannelChanged);
00257     
00258   while(!terminated)
00259     {
00260       msecsleep(10);
00261 
00262       double raw, src_raw = 0, src_temp = 0, temp;
00263       clearRaw();
00264       XTime time_awared = XTime::now();
00265       // try/catch exception of communication errors
00266       try {
00267           shared_ptr<XChannel> src_ch = *m_currentChannel;
00268           if(src_ch)
00269             {
00270               shared_ptr<XThermometer> thermo = *src_ch->thermometer();
00271               src_raw = getRaw(src_ch);
00272               src_temp = (!thermo) ? src_raw : thermo->getTemp(src_raw);
00273               m_sourceTemp->value(src_temp);
00274             }
00275           atomic_shared_ptr<const XNode::NodeList> list(m_channels->children());
00276           if(list) { 
00277               unsigned int idx = 0;
00278               for(XNode::NodeList::const_iterator it = list->begin(); it != list->end(); it++) {
00279                 shared_ptr<XChannel> ch = dynamic_pointer_cast<XChannel>(*it);
00280                   if(src_ch == ch)
00281                     {
00282                       temp = src_temp;
00283                       raw = src_raw;
00284                     }
00285                   else
00286                     {
00287                       if(!m_multiread) continue;
00288                       shared_ptr<XThermometer> thermo = *ch->thermometer();
00289                       raw = getRaw(ch);
00290                       temp = (!thermo) ? raw : thermo->getTemp(raw);
00291                     }
00292                   push((unsigned short)idx);
00293                   push((unsigned short)0); // reserve
00294                   push(float(raw));
00295                   push(float(temp));
00296                   idx++;
00297                 }
00298           }
00299             
00300           //calicurate std. deviations in some periods
00301           double tau = *interval() * 4.0;
00302           if(tau <= 1) tau = 4.0;
00303           XTime newtime = XTime::now();
00304           double dt = newtime - lasttime;
00305           lasttime = newtime;
00306           double terr = src_temp - *targetTemp();
00307           terr = terr * terr;
00308           tempAvg = (tempAvg - temp) * exp(-dt / tau) + temp;
00309           tempErrAvg = (tempErrAvg - terr) * exp(-dt / tau) + terr;
00310           stabilized()->value(sqrt(tempErrAvg)); //stderr
00311           heaterPower()->value(getHeater());
00312       }
00313       catch (XKameError &e) {

00315           continue;
00316       }
00317       finishWritingRaw(time_awared, XTime::now());
00318     }

00325   m_lsnOnTargetTempChanged.reset();

00327   m_lsnOnHeaterModeChanged.reset();
