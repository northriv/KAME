00127         unsigned int t = context->t;
00128         unsigned int p = context->p;
00129         std::complex<double> x = 0.0, y = 0.0;
00130         for(unsigned int i = p + 1; i < t; i++) {
00131                 x += context->epsilon[i] * std::conj(context->eta[i-1]);
00132                 y += std::norm(context->epsilon[i]) + std::norm(context->eta[i-1]);
00133         }
00134         std::complex<double> alpha = x / y;
00135         alpha *= -2;
00136         for(unsigned int i = t - 1; i >= p + 1; i--) {
00137                 context->eta[i] = context->eta[i-1] + std::conj(alpha) * context->epsilon[i];
00138                 context->epsilon[i] += alpha * context->eta[i-1];
00139         }
00140         std::vector<std::complex<double> > a_next(p + 2);
00141         for(unsigned int i = 0; i < p + 2; i++) {
00142                 a_next[i] = context->a[i] + alpha * std::conj(context->a[p + 1 - i]);
00143         }
00144         std::copy(a_next.begin(), a_next.end(), context->a.begin());
00145         context->sigma2 *= 1 - std::norm(alpha);
