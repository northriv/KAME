00147 XNMRFSpectrum::analyze(const shared_ptr<XDriver> &emitter) throw (XRecordError&)
00148 {

00150   ASSERT( _sg1 );
00151   ASSERT( _sg1->time() );

00154     if(_sg2) {
00155         if(fabs(freq - (_sg2->freqRecorded() - *sg2FreqOffset())) > freq * 1e-6)
00156             throw XRecordError(KAME::i18n("Conflicting freqs of 2 SGs."), __FILE__, __LINE__); 
00157     }
00158     
00159   shared_ptr<XNMRPulseAnalyzer> _pulse = *pulse();
00160   ASSERT( _pulse );
00161   ASSERT( _pulse->time() );
00162   ASSERT( emitter != _sg1 );
00163   ASSERT( emitter != _sg2 );
00164    
00165   double cfreq = *centerFreq(); //MHz
00166   double freq_span = *freqSpan() * 1e-3; //MHz
00167   double freq_step = *freqStep() * 1e-3; //MHz
00168   if(cfreq <= freq_span/2) {
00169     throw XRecordError(KAME::i18n("Invalid center freq."), __FILE__, __LINE__);
00170   }
00171   if(freq_span <= freq_step*2) {
00172     throw XRecordError(KAME::i18n("Too large freq. step."), __FILE__, __LINE__);
00173   }
00174 
00175   bool clear = (m_timeClearRequested > _pulse->timeAwared());
00176   
00177   int len = _pulse->ftWave().size();
00178   double _df = _pulse->dFreq() * 1e-6; //MHz
00179   if((len == 0) || (_df == 0)) {
00180     throw XRecordError(KAME::i18n("Invalid waveform."), __FILE__, __LINE__);  
00181   }
00182   
00183   double freq_min = cfreq - freq_span/2;
00184   double freq_max = cfreq + freq_span/2;
00185 
00186   if((fabs(df() - _df) > 1e-6) || clear)
00187     {
00188       m_df = _df;
00189       m_wave.clear();
00190       m_counts.clear();
00191     }
00192   else {
00193     for(int i = 0; i < rint(m_fMin / df()) - rint(freq_min / df()); i++) {
00194          m_wave.push_front(0.0);
00195          m_counts.push_front(0);
00196     }
00197     for(int i = 0; i < rint(freq_min / df()) - rint(m_fMin / df()); i++) {
00198          if(!m_wave.empty()) {
00199              m_wave.pop_front();
00200              m_counts.pop_front();
00201          }
00202     }
00203   }
00204   m_fMin = freq_min;
00205   int length = lrint(freq_span / df());
00206   m_wave.resize(length, 0.0);
00207   m_counts.resize(length, 0);
00208 
00209   if(clear) {
00210     m_spectrum->clear();
00211         throw XSkippedRecordError(__FILE__, __LINE__);
00212   }
00213   if(emitter != _pulse) throw XSkippedRecordError(__FILE__, __LINE__);
00214   
00215   int bw = lrint(*bandWidth() * 1e-3 / df());
00216   for(int i = std::max(0, (len - bw) / 2); i < std::min(len, (len + bw) / 2); i++)
00217     {
00218       double f = (i - len/2) * df(); //MHz
00219         add(freq + f, _pulse->ftWave()[i]);
00220     }
00221

00224    {
00225       double newf = freq + freq_step; //MHz
00226     
00227       if(_sg1) _sg1->freq()->value(newf + *sg1FreqOffset());
00228       if(_sg2) _sg2->freq()->value(newf + *sg2FreqOffset());
