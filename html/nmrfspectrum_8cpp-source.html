00129 XNMRFSpectrum::afterFSSum() {
00130         double freq = getCurrentCenterFreq() * 1e-6;
00131         //set new freq
00132         if(*active())
00133         {
00134             shared_ptr<XSG> _sg1 = *sg1();
00135                 ASSERT( _sg1 );
00136                 ASSERT( _sg1->time() );
00137             shared_ptr<XSG> _sg2 = *sg2();
00138 
00139             double cfreq = *centerFreq(); //MHz
00140                 double freq_span = *freqSpan() * 1e-3; //MHz
00141                 double freq_step = *freqStep() * 1e-3; //MHz
00142                 if(cfreq <= freq_span/2) {
00143                         throw XRecordError(KAME::i18n("Invalid center freq."), __FILE__, __LINE__);
00144                 }
00145                 if(freq_span <= freq_step*2) {
00146                         throw XRecordError(KAME::i18n("Too large freq. step."), __FILE__, __LINE__);
00147                 }
00148           
00149                 if(_sg1) unlockConnection(_sg1);
00150                 if(_sg2) unlockConnection(_sg2);
00151 
00152                 double newf = freq + freq_step; //MHz
00153     
00154                 if(_sg1) _sg1->freq()->value(newf + *sg1FreqOffset());
00155                 if(_sg2) _sg2->freq()->value(newf + *sg2FreqOffset());
00156                 if(newf >= getMaxFreq() * 1e-6)
00157                         active()->value(false);
00158         }       
00159 }
