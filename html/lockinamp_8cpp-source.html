00167 void *
00168 XLIA::execute(const atomic<bool> &terminated)
00169 {
00170 
00171   m_lsnOutput = output()->onValueChanged().connectWeak(
00172                           shared_from_this(), &XLIA::onOutputChanged);
00173   m_lsnFreq = frequency()->onValueChanged().connectWeak(
00174                           shared_from_this(), &XLIA::onFreqChanged);
00175   m_lsnSens = sensitivity()->onValueChanged().connectWeak(
00176                         shared_from_this(), &XLIA::onSensitivityChanged);
00177   m_lsnTimeConst = timeConst()->onValueChanged().connectWeak(
00178                          shared_from_this(), &XLIA::onTimeConstChanged);
00179 
00180   while(!terminated)
00181     {
00182       double fetch_freq = *fetchFreq();
00183       double wait = 0;
00184       if(fetch_freq > 0) {
00185          sscanf(timeConst()->to_str().c_str(), "%lf", &wait);
00186          wait *= 1000.0 / fetch_freq;
00187       }
00188       if(wait > 0) msecsleep(lrint(wait));
00189       
00190       double x, y;
00191       XTime time_awared = XTime::now();
00192       // try/catch exception of communication errors
00193       try {
00194           get(&x, &y);
00195       }
00196       catch (XKameError &e) {
00197           e.print(getLabel() + " " + KAME::i18n("Read Error, "));
00198           continue;
00199       }
00200       clearRaw();
00201       push(x);
00202       push(y);
00203       finishWritingRaw(time_awared, XTime::now());
00204     }
00205   
00206   m_lsnOutput.reset();
00207   m_lsnFreq.reset();
00208   m_lsnSens.reset();
00209   m_lsnTimeConst.reset();
00210 
00211   afterStop();
