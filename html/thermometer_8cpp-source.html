00053 XLakeShore::getRawValue(double temp) const
00054 {
00055   //using Newton's method
00056   double x, y, dypdx, val;
00057   if(temp < *tempMin()) return *resMax();
00058   if(temp > *tempMax()) return *resMin();
00059   //    x = (log10(RMin) + log10(RMax)) / 2;
00060   val = *resMin();
00061   for(double dy = 0.0001;;dy *= 2)
00062     {
00063       if(dy > 1.0) return *resMin();
00064       double t = (double)KAME::rand() / (RAND_MAX - 1);
00065       x = (log10(*resMax()) * t + log10(*resMin()) * (1 - t));
00066       for(int i = 0; i < 100; i++)
00067         {
00068           y = getTemp(pow(10, x)) - temp;
00069           if(fabs(y) < dy)
00070             {
00071               val = pow(10, x);
00072               return val;
00073             }
00074           dypdx = (y - (getTemp(pow(10, x - 0.00001)) - temp)) / 0.00001;
00075           if(dypdx != 0) x -= y / dypdx;
00076           if((x > log10(*resMax())) || (x < log10(*resMin())) || (dypdx == 0))
00077             {
00078               double t = (double)KAME::rand() / (RAND_MAX - 1);
00079               x = (log10(*resMax()) * t + log10(*resMin()) * (1 - t));
00080             }
00081         }
00082     }
00083   return val;

00087 XLakeShore::getTemp(double res) const
00088 {
00089   double temp = 0, z, u = 0;
00090   if(res > *resMax()) return *tempMin();
00091   if(res < *resMin()) return *tempMax();
00092   z = log10(res);
00093   unsigned int n;
00094   atomic_shared_ptr<const XNode::NodeList> zu_list(zu()->children());
00095   if(!zu_list) return 0;
00096   atomic_shared_ptr<const XNode::NodeList> zl_list(zl()->children());
00097   if(!zl_list) return 0;
00098   for(n = 0; n < zu_list->size(); n++)
00099     {
00100         double zu = *dynamic_pointer_cast<XDoubleNode>(zu_list->at(n));
00101         double zl = *dynamic_pointer_cast<XDoubleNode>(zl_list->at(n));
00102       u = (z - zu + z - zl) / (zu - zl);
00103       if((u >= -1) && (u <= 1)) break;
00104     }
00105   if(n >= zu_list->size())
00106     return 0;
00107   atomic_shared_ptr<const XNode::NodeList> ai_list(ai()->children());
00108   if(!ai_list) return 0;
00109   atomic_shared_ptr<const XNode::NodeList> ai_n_list(ai_list->at(n)->children());
00110   if(!ai_n_list) return 0;
00111   for(unsigned int i = 0; i < ai_n_list->size(); i++)
00112     {
00113         double ai_n_i = *dynamic_pointer_cast<XDoubleNode>(ai_n_list->at(i));
00114         temp += ai_n_i * cos(i * acos(u));
00115     }
00116   return temp;
