00943 XGraph::GFloat

00945 {
00946   XGraph::GFloat pos;
00947   if(m_bLogscaleFixed)
00948     {

00950     return - XGraph::GFLOAT_MAX;

00952             m_invLogMaxOverMinFixed = 1 / VFLOAT_LOG(m_maxFixed / m_minFixed);  
00953       pos = VFLOAT_LOG(x / m_minFixed) * m_invLogMaxOverMinFixed;
00954     }

00956     {
00957       if(m_maxFixed <= m_minFixed) return -1;
00958       if(m_invMaxMinusMinFixed < 0)
00959          m_invMaxMinusMinFixed = 1 / (m_maxFixed - m_minFixed);
00960       pos = (x - m_minFixed) * m_invMaxMinusMinFixed;
00961     }
00962   return pos;

01024 XAxis::Tic
01025 XAxis::queryTic(int len, int pos, XGraph::VFloat *ticnum)
01026 {
01027   XGraph::VFloat x, t;
01028   if(m_bLogscaleFixed)
01029     {
01030       x = axisToVal((XGraph::GFloat)pos / len);
01031       if(x <= 0) return NoTics;
01032       t = VFLOAT_POW((XGraph::VFloat)10.0, VFLOAT_RINT(VFLOAT_LOG10(x)));
01033       if(GFLOAT_LRINT(valToAxis(t) * len) == pos)
01034     {
01035       *ticnum = t;
01036       return MajorTic;
01037     }
01038       x = x / t;
01039       if(x < 1)
01040     t = VFLOAT_RINT(x / (XGraph::VFloat)0.1) * (XGraph::VFloat)0.1 * t;
01041       else
01042     t = VFLOAT_RINT(x) * t;
01043       if(GFLOAT_LRINT(valToAxis(t) * len) == pos)
01044     {
01045       *ticnum = t;
01046       return MinorTic;
01047     }
01048       return NoTics;
01049     }
01050   else
01051     {
01052       x = axisToVal((XGraph::GFloat)pos / len);
01053       t = VFLOAT_RINT(x / m_majorFixed) * m_majorFixed;
01054       if(GFLOAT_LRINT(valToAxis(t) * len) == pos)
01055     {
01056       *ticnum = t;
01057       return MajorTic;
01058     }
01059       t = VFLOAT_RINT(x / m_minorFixed) * m_minorFixed;
01060       if(GFLOAT_LRINT(valToAxis(t) * len) == pos)
01061     {
01062       *ticnum = t;
01063       return MinorTic;
01064     }
01065       return NoTics;
01066     }
