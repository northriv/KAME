00546   XGraph::GPoint g1 = c1.graph;

00549   XGraph::GFloat idx = 1.0 / (g1.x - g2.x);
00550   XGraph::GFloat tx0 = -g2.x * idx;

00557   XGraph::GFloat ty1 = (1.0 - g2.y) * idy;

00574 
00575   if(tmin >= tmax) return false;
00576   if(tmin >= 1.0) return false;

00578    
00579    if(tmin > 0.0) {

00581                 tmin*g1.x + (1.0 - tmin) *g2.x,
00582                 tmin*g1.y + (1.0 - tmin) *g2.y,
00583                 tmin*g1.z + (1.0 - tmin) *g2.z)
00584                 , s2);

00887 {
00888   XGraph::GFloat pos;
00889   if(m_bLogscaleFixed)
00890     {

00892         return - XGraph::GFLOAT_MAX;
00893       if(m_invLogMaxOverMinFixed < 0)
00894                 m_invLogMaxOverMinFixed = 1 / VFLOAT_LOG(m_maxFixed / m_minFixed);      
00895       pos = VFLOAT_LOG(x / m_minFixed) * m_invLogMaxOverMinFixed;
00896     }
00897   else
00898     {
00899       if(m_maxFixed <= m_minFixed) return -1;
00900       if(m_invMaxMinusMinFixed < 0)
00901          m_invMaxMinusMinFixed = 1 / (m_maxFixed - m_minFixed);
00902       pos = (x - m_minFixed) * m_invMaxMinusMinFixed;
00903     }
00904   return pos;

00968 XAxis::Tic
00969 XAxis::queryTic(int len, int pos, XGraph::VFloat *ticnum)
00970 {
00971   XGraph::VFloat x, t;
00972   if(m_bLogscaleFixed)
00973     {
00974       x = axisToVal((XGraph::GFloat)pos / len);
00975       if(x <= 0) return NoTics;
00976       t = VFLOAT_POW((XGraph::VFloat)10.0, VFLOAT_RINT(VFLOAT_LOG10(x)));
00977       if(GFLOAT_LRINT(valToAxis(t) * len) == pos)
00978         {
00979           *ticnum = t;
00980           return MajorTic;
00981         }
00982       x = x / t;
00983       if(x < 1)
00984         t = VFLOAT_RINT(x / (XGraph::VFloat)0.1) * (XGraph::VFloat)0.1 * t;
00985       else
00986         t = VFLOAT_RINT(x) * t;
00987       if(GFLOAT_LRINT(valToAxis(t) * len) == pos)
00988         {
00989           *ticnum = t;
00990           return MinorTic;
00991         }
00992       return NoTics;
00993     }
00994   else
00995     {
00996       x = axisToVal((XGraph::GFloat)pos / len);
00997       t = VFLOAT_RINT(x / m_majorFixed) * m_majorFixed;
00998       if(GFLOAT_LRINT(valToAxis(t) * len) == pos)
00999         {
01000           *ticnum = t;
01001           return MajorTic;
01002         }
01003       t = VFLOAT_RINT(x / m_minorFixed) * m_minorFixed;
01004       if(GFLOAT_LRINT(valToAxis(t) * len) == pos)
01005         {
01006           *ticnum = t;
01007           return MinorTic;
01008         }
01009       return NoTics;
01010     }
01011 }

01057 int
01058 XAxis::drawAxis(XQGraphPainter *painter)
01059 {
01060   XGraph::SFloat LenMajorTicL = 0.01;
01061   XGraph::SFloat LenMinorTicL = 0.005;
01062 
01063   painter->setColor(*ticColor());
01064 
01065   XGraph::ScrPoint s1, s2;
01066   axisToScreen(0.0, &s1);
01067   axisToScreen(1.0, &s2);
01068 
01069   painter->beginLine();
01070   painter->setVertex(s1);
01071   painter->setVertex(s2);
01072   painter->endLine();
01073   
01074   if(*displayLabel())
01075     {
01076         drawLabel(painter);
01077         
01078     }
01079   if(m_bLogscaleFixed && (m_minFixed < 0)) return -1;
01080   if(m_maxFixed <= m_minFixed) return -1;
01081   
01082   int len = SFLOAT_LRINT(*length() / painter->resScreen());
01083   painter->defaultFont();
01084   XGraph::GFloat mindx = 2, lastg = -1;
01086   for(int i = 0; i < len; i++) {
01087     XGraph::VFloat z;
01088     XGraph::GFloat x = (XGraph::GFloat)i / len;
01089       if(queryTic(len, i, &z) == MajorTic)
01090         {
01091                 if(mindx > x - lastg) {
01092                         axisToScreen(x, &s1);
01093                         s2 = s1;
01094                         XGraph::ScrPoint s3;
01095                         axisToScreen(lastg, &s3);
01096                         s3 -= s2;
01097                         s3 *= 0.7;
01098                         painter->posOffAxis(m_dirVector, &s1, AxisToTicLabel);
01099                         s2 -= s1;
01100                         s2 *= -1;
01101                         
01102                         double var = setprec(z, m_bLogscaleFixed ? (XGraph::VFloat)z :  m_minorFixed);
01103                         painter->selectFont(valToString(var), s1, s2, s3, 0);
01104                         
01105                         mindx = x - lastg;
01106                 }
01107                 lastg = x;
01108         }
01109   }
01110   
01111   for(int i = 0; i < len; i++) {
01112     XGraph::VFloat z;
01113     XGraph::GFloat x = (XGraph::GFloat)i / len;
01114       switch(queryTic(len, i, &z))
01115         {
01116         case MajorTic:
01117           if(*displayMajorTics())
01118             {
01119               axisToScreen(x, &s1);
01120               painter->posOffAxis(m_dirVector, &s1, LenMajorTicL);
01121               axisToScreen(x, &s2);
01122               painter->posOffAxis(m_dirVector, &s2, -LenMajorTicL);
01123               painter->setColor(*ticColor());
01124               painter->beginLine(1.0);
01125               painter->setVertex(s1);
01126               painter->setVertex(s2);
01127               painter->endLine();
01128             }
01129           if(*displayTicLabels())
01130             {
01131               axisToScreen(x, &s1);
01132               painter->posOffAxis(m_dirVector, &s1, AxisToTicLabel);
01133               double var = setprec(z, m_bLogscaleFixed ? (XGraph::VFloat)z : m_minorFixed);
01134               painter->drawText(s1, valToString(var));
01135             }
01136           break;
01137         case MinorTic:
