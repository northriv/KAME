01011 XAxis::Tic
01012 XAxis::queryTic(int len, int pos, XGraph::VFloat *ticnum)
01013 {
01014   XGraph::VFloat x, t;
01015   if(m_bLogscaleFixed)
01016     {
01017       x = axisToVal((XGraph::GFloat)pos / len);
01018       if(x <= 0) return NoTics;
01019       t = VFLOAT_POW((XGraph::VFloat)10.0, VFLOAT_RINT(VFLOAT_LOG10(x)));
01020       if(GFLOAT_LRINT(valToAxis(t) * len) == pos)
01021     {
01022       *ticnum = t;
01023       return MajorTic;
01024     }
01025       x = x / t;
01026       if(x < 1)
01027     t = VFLOAT_RINT(x / (XGraph::VFloat)0.1) * (XGraph::VFloat)0.1 * t;
01028       else
01029     t = VFLOAT_RINT(x) * t;
01030       if(GFLOAT_LRINT(valToAxis(t) * len) == pos)
01031     {
01032       *ticnum = t;
01033       return MinorTic;
01034     }
01035       return NoTics;
01036     }
01037   else
01038     {
01039       x = axisToVal((XGraph::GFloat)pos / len);
01040       t = VFLOAT_RINT(x / m_majorFixed) * m_majorFixed;
01041       if(GFLOAT_LRINT(valToAxis(t) * len) == pos)
01042     {
01043       *ticnum = t;
01044       return MajorTic;
01045     }
01046       t = VFLOAT_RINT(x / m_minorFixed) * m_minorFixed;
01047       if(GFLOAT_LRINT(valToAxis(t) * len) == pos)
01048     {
01049       *ticnum = t;
01050       return MinorTic;
01051     }
01052       return NoTics;
01053     }
