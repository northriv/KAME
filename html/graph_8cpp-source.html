00543         *s1 = c1.scr; *s2 = c2.scr;

00551   XGraph::GPoint g2 = c2.graph;
00552   
00553   XGraph::GFloat idx = 1.0 / (g1.x - g2.x);
00554   XGraph::GFloat tx0 = -g2.x * idx;
00555   XGraph::GFloat tx1 = (1.0 - g2.x) * idx;

00562   XGraph::GFloat tymin = min(ty0, ty1);
00563   XGraph::GFloat tymax = max(ty0, ty1);
00564

00898   if(m_bLogscaleFixed)

00905     }
00906   else
00907     {
00908       if(m_maxFixed <= m_minFixed) return -1;
00909       if(m_invMaxMinusMinFixed < 0)
00910          m_invMaxMinusMinFixed = 1 / (m_maxFixed - m_minFixed);
00911       pos = (x - m_minFixed) * m_invMaxMinusMinFixed;
00912     }
00913   return pos;

00977 XAxis::Tic
00978 XAxis::queryTic(int len, int pos, XGraph::VFloat *ticnum)
00979 {
00980   XGraph::VFloat x, t;
00981   if(m_bLogscaleFixed)
00982     {
00983       x = axisToVal((XGraph::GFloat)pos / len);
00984       if(x <= 0) return NoTics;
00985       t = VFLOAT_POW((XGraph::VFloat)10.0, VFLOAT_RINT(VFLOAT_LOG10(x)));
00986       if(GFLOAT_LRINT(valToAxis(t) * len) == pos)
00987         {
00988           *ticnum = t;
00989           return MajorTic;
00990         }
00991       x = x / t;
00992       if(x < 1)
00993         t = VFLOAT_RINT(x / (XGraph::VFloat)0.1) * (XGraph::VFloat)0.1 * t;
00994       else
00995         t = VFLOAT_RINT(x) * t;
00996       if(GFLOAT_LRINT(valToAxis(t) * len) == pos)
00997         {
00998           *ticnum = t;
00999           return MinorTic;
01000         }
01001       return NoTics;
01002     }
01003   else
01004     {
01005       x = axisToVal((XGraph::GFloat)pos / len);
01006       t = VFLOAT_RINT(x / m_majorFixed) * m_majorFixed;
01007       if(GFLOAT_LRINT(valToAxis(t) * len) == pos)
01008         {
01009           *ticnum = t;
01010           return MajorTic;
01011         }
01012       t = VFLOAT_RINT(x / m_minorFixed) * m_minorFixed;
01013       if(GFLOAT_LRINT(valToAxis(t) * len) == pos)
01014         {
01015           *ticnum = t;
01016           return MinorTic;
01017         }
01018       return NoTics;
01019     }
01020 }

01066 int
01067 XAxis::drawAxis(XQGraphPainter *painter)
01068 {

01070   XGraph::SFloat LenMinorTicL = 0.005;
01071

01075   axisToScreen(0.0, &s1);
01076   axisToScreen(1.0, &s2);
01077

01079   painter->setVertex(s1);
01080   painter->setVertex(s2);
01081   painter->endLine();
01082   
01083   if(*displayLabel())
01084     {
01085         drawLabel(painter);
01086         
01087     }
01088   if(m_bLogscaleFixed && (m_minFixed < 0)) return -1;
01089   if(m_maxFixed <= m_minFixed) return -1;
01090   
01091   int len = SFLOAT_LRINT(*length() / painter->resScreen());
01092   painter->defaultFont();
01093   XGraph::GFloat mindx = 2, lastg = -1;
01095   for(int i = 0; i < len; i++) {
01096     XGraph::VFloat z;
01097     XGraph::GFloat x = (XGraph::GFloat)i / len;
01098       if(queryTic(len, i, &z) == MajorTic)
01099         {
01100                 if(mindx > x - lastg) {
01101                         axisToScreen(x, &s1);
01102                         s2 = s1;
01103                         XGraph::ScrPoint s3;
01104                         axisToScreen(lastg, &s3);
01105                         s3 -= s2;
01106                         s3 *= 0.7;
01107                         painter->posOffAxis(m_dirVector, &s1, AxisToTicLabel);
01108                         s2 -= s1;
01109                         s2 *= -1;
01110                         
01111                         double var = setprec(z, m_bLogscaleFixed ? (XGraph::VFloat)z :  m_minorFixed);
01112                         painter->selectFont(valToString(var), s1, s2, s3, 0);
01113                         
01114                         mindx = x - lastg;
01115                 }
01116                 lastg = x;
01117         }
01118   }
01119   
01120   for(int i = 0; i < len; i++) {
01121     XGraph::VFloat z;
01122     XGraph::GFloat x = (XGraph::GFloat)i / len;
01123       switch(queryTic(len, i, &z))
01124         {
01125         case MajorTic:
01126           if(*displayMajorTics())
01127             {
01128               axisToScreen(x, &s1);
01129               painter->posOffAxis(m_dirVector, &s1, LenMajorTicL);
01130               axisToScreen(x, &s2);
01131               painter->posOffAxis(m_dirVector, &s2, -LenMajorTicL);
01132               painter->setColor(*ticColor());
01133               painter->beginLine(1.0);
01134               painter->setVertex(s1);
01135               painter->setVertex(s2);
01136               painter->endLine();
01137             }
01138           if(*displayTicLabels())
01139             {
01140               axisToScreen(x, &s1);
01141               painter->posOffAxis(m_dirVector, &s1, AxisToTicLabel);
01142               double var = setprec(z, m_bLogscaleFixed ? (XGraph::VFloat)z : m_minorFixed);
01143               painter->drawText(s1, valToString(var));
01144             }
01145           break;
01146         case MinorTic:
01147           if(*displayMinorTics())
01148             {
01149               axisToScreen(x, &s1);
01150               painter->posOffAxis(m_dirVector, &s1, LenMinorTicL);
