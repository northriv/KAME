00047 double XNMRPulseAnalyzer::windowFuncHanning(double x) {
00048         if(fabs(x) >= 0.5) return 0.0;
00049         return 0.5 + 0.5*cos(2*PI*x);
00050 }
00051 double XNMRPulseAnalyzer::windowFuncHamming(double x) {
00052         if(fabs(x) >= 0.5) return 0.0;
00053         return 0.54 + 0.46*cos(2*PI*x);
00054 }
00055 double XNMRPulseAnalyzer::windowFuncBlackman(double x) {
00056         if(fabs(x) >= 0.5) return 0.0;
00057         return 0.42323+0.49755*cos(2*PI*x)+0.07922*cos(4*PI*x);
00058 }
00059 double XNMRPulseAnalyzer::windowFuncBlackmanHarris(double x) {
00060         if(fabs(x) >= 0.5) return 0.0;
00061         return 0.35875+0.48829*cos(2*PI*x)+0.14128*cos(4*PI*x)+0.01168*cos(6*PI*x);
00062 }
00063 double XNMRPulseAnalyzer::windowFuncFlatTop(double x) {
00064         return windowFuncHamming(x)*((fabs(x) < 1e-4) ? 1 : sin(4*PI*x)/(4*PI*x));
00065 }
00066 double XNMRPulseAnalyzer::windowFuncKaiser(double x, double alpha) {
00067         if(fabs(x) >= 0.5) return 0.0;
00068         x *= 2;
00069         x = sqrt(std::max(1 - x*x, 0.0));
00070         return bessel_i0(PI*alpha*x) / bessel_i0(PI*alpha);
00071 }
00072 double XNMRPulseAnalyzer::windowFuncKaiser1(double x) {
00073         return windowFuncKaiser(x, 3.0);
00074 }
00075 double XNMRPulseAnalyzer::windowFuncKaiser2(double x) {
00076         return windowFuncKaiser(x, 7.2);
00077 }
00078 double XNMRPulseAnalyzer::windowFuncKaiser3(double x) {
00079         return windowFuncKaiser(x, 15.0);

00247 XNMRPulseAnalyzer::backgroundSub(const std::deque<std::complex<double> > &wave,
00248      int pos, int length, int bgpos, int bglength, twindowfunc windowfunc)
00249 {
00250         if(*useDNR() && (bglength > 0))
00251     {
00252         const int dnrlen = bglength;
00253         const int dnrpos = bgpos;
00254 
00255       //FFT plan for subtraction
00256     if(m_dnrsubfftlen != dnrlen)
00257     {
00258       if(m_dnrsubfftlen >= 0) fftw_destroy_plan(m_dnrsubfftplan);
00259       m_dnrsubfftlen = dnrlen;
00260       m_dnrsubfftin.resize(m_dnrsubfftlen);
00261       m_dnrsubfftout.resize(m_dnrsubfftlen);
00262       m_dnrsubfftplan = fftw_create_plan(m_dnrsubfftlen, FFTW_FORWARD, FFTW_ESTIMATE);
00263     }
00264     //IFFT plan for the subtracted
00265     if(m_dnrpulsefftlen != std::max(dnrlen, length))
00266     {
00267       if(m_dnrpulsefftlen >= 0)  fftw_destroy_plan(m_dnrpulsefftplan);
00268       m_dnrpulsefftlen = std::max(dnrlen, length);
00269       m_dnrpulsefftin.resize(m_dnrpulsefftlen);
00270       m_dnrpulsefftout.resize(m_dnrpulsefftlen);
00271       m_dnrpulsefftplan = fftw_create_plan(m_dnrpulsefftlen, FFTW_BACKWARD, FFTW_ESTIMATE);
00272     }
00273 
00274     for(int i = 0; i < m_dnrsubfftlen; i++)
00275     {
00276       m_dnrsubfftin[i].re = std::real(wave[i + pos + dnrpos]);
00277       m_dnrsubfftin[i].im = std::imag(wave[i + pos + dnrpos]);
00278     }
00279     fftw_one(m_dnrsubfftplan, &m_dnrsubfftin[0], &m_dnrsubfftout[0]);
00280 
00281     //calcurate noise level
00282         double nlevel = 0.0;
00283         int nlevel_cnt = 0;
00284 #define NOISE_BW 20
00285         for(int i = 0; i < m_dnrsubfftlen; i++)
00286     {
00287       int k = (i < m_dnrsubfftlen/2) ? i : (i - m_dnrsubfftlen);
00288       if((abs(k) < NOISE_BW) && (k != 0))
00289         { 
00290           nlevel += sqrt(m_dnrsubfftout[i].re * m_dnrsubfftout[i].re
00291                  + m_dnrsubfftout[i].im * m_dnrsubfftout[i].im);
00292           nlevel_cnt++;
00293         }
00294     }
00295     nlevel /= nlevel_cnt;
00296     nlevel *= 2.0; //Threshold value
00297 
00298     for(int i = 0; i < m_dnrpulsefftlen; i++)
00299     {
00300       m_dnrpulsefftin[i].re = 0.0;
00301       m_dnrpulsefftin[i].im = 0.0;
00302     }        
00303 
00304     m_noisePower = 0.0;
00305     for(int i = 0; i < m_dnrsubfftlen; i++)
00306     {
00307       int k = (i < m_dnrsubfftlen/2) ? i : (i - m_dnrsubfftlen);
00308       if((k != 0)
00309          && (sqrt(m_dnrsubfftout[i].re * m_dnrsubfftout[i].re + m_dnrsubfftout[i].im * m_dnrsubfftout[i].im) < nlevel))
00310           {
00311           m_noisePower += m_dnrsubfftout[i].re * m_dnrsubfftout[i].re 
00312                     + m_dnrsubfftout[i].im * m_dnrsubfftout[i].im;
00313           continue;            
00314       }
00315       const double dphase = - 2 * PI * dnrpos * i / m_dnrsubfftlen;
00316 
00317       const double cx = cos(dphase) / m_dnrsubfftlen;
00318       const double sx = sin(dphase) / m_dnrsubfftlen;
00319       const int j = i * (m_dnrpulsefftlen - 1) / (m_dnrsubfftlen - 1);
00320       ASSERT((j >= 0) || (j < m_dnrpulsefftlen));
00321       m_dnrpulsefftin[j].re += cx * m_dnrsubfftout[i].re - sx * m_dnrsubfftout[i].im;
00322       m_dnrpulsefftin[j].im += cx * m_dnrsubfftout[i].im + sx * m_dnrsubfftout[i].re;
00323     }
00324     m_noisePower /= bglength*bglength;   
00325 
00326         fftw_one(m_dnrpulsefftplan, &m_dnrpulsefftin[0], &m_dnrpulsefftout[0]);
00327     for(int i = 0; i < length; i++)
00328     {
00329       const int j = i % m_dnrpulsefftlen;
00330       std::complex<double> c(m_dnrpulsefftout[j].re, m_dnrpulsefftout[j].im);
00331       m_wave[i] = wave[pos + i] - c;
00332 //    WaveRe[i] = dnrpulsefftout[j].re;
00333 //    WaveIm[i] = dnrpulsefftout[j].im;
00334     }
00335   }
00336   else
00337   {
00338       std::complex<double> bg = 0;
00339       m_noisePower = 1.0;
00340       if(bglength)
00341     {
00342       double normalize = 0.0;
00343           for(int i = 0; i < bglength; i++)
00344         {
00345         double z = windowfunc( (double)i / bglength - 0.5);
00346           bg += z * wave[pos + i + bgpos];
00347           normalize += z;
00348         }
00349           bg /= normalize;
00350 
00351           m_noisePower = 0.0;
00352           for(int i = bgpos; i < bgpos + bglength; i++)
00353         {
00354               m_noisePower += std::norm(wave[pos + i] - bg);
00355         }
00356           m_noisePower /= bglength;
00357     }
00358 
00359     for(int i = 0; i < length; i++)
00360     {
00361       m_wave[i] = wave[pos + i] - bg;
00362     }
00363   }            
00364 }
00365 void
00366 XNMRPulseAnalyzer::rotNFFT(int ftpos, 
00367     double ph, std::deque<std::complex<double> > &wave, 
00368     std::deque<std::complex<double> > &ftwave, twindowfunc windowfunc, int diffreq)
00369 {
00370 const int length = wave.size();
00371   //phase advance
00372   std::complex<double> cph(cos(ph), sin(ph));
00373   for(int i = 0; i < length; i++)
00374     {
00375       wave[i] *= cph;
00376     }
00377 
00378   //fft
00379   for(int i = 0; i < m_fftlen; i++)
00380     {
00381       const int j = (ftpos + i >= m_fftlen) ? (ftpos + i - m_fftlen) : (ftpos + i);
00382       const double z = windowfunc((j - ftpos) / (double)length);
00383       if((j >= length) || (j < 0)) {
00384             m_fftin[i].re = 0;
00385             m_fftin[i].im = 0;
00386       }
00387       else {
00388          wave[j] *= z;
00389          m_fftin[i].re = std::real(wave[j]);
00390          m_fftin[i].im = std::imag(wave[j]);
00391       }
00392     }
00393     
00394   fftw_one(m_fftplan, &m_fftin[0], &m_fftout[0]);
00395 
00396   for(int i = 0; i < m_fftlen; i++)
00397     {
00398       const int k = i + diffreq;
00399       if((k >= 0) && (k < m_fftlen)) {
00400           const int j = (k < m_fftlen / 2) ? (m_fftlen / 2 + k) : (k - m_fftlen / 2);
00401           const double normalize = 1.0 / length;
00402           ftwave[i] = std::complex<double>(m_fftout[j].re * normalize, m_fftout[j].im * normalize);
00403       }
00404       else {
00405          ftwave[i] = 0;
00406       }
00407     }
00408 }

00432 XNMRPulseAnalyzer::analyze(const shared_ptr<XDriver> &) throw (XRecordError&)
00433 {
00434   const shared_ptr<XDSO> _dso = *dso();
00435   ASSERT( _dso );
00436   ASSERT( _dso->time() );
00437   
00438   if(_dso->numChannelsRecorded() < 1) {
00439     throw XRecordError(KAME::i18n("No record in DSO"), __FILE__, __LINE__);
00440   }
00441   if(_dso->numChannelsRecorded() < 2) {
00442     throw XRecordError(KAME::i18n("Two channels needed in DSO"), __FILE__, __LINE__);
00443   }
00444 
00445   const double interval = _dso->timeIntervalRecorded();
00446   
00447   if(interval <= 0) {
00448     throw XRecordError(KAME::i18n("Invalid time interval in waveforms."), __FILE__, __LINE__);
00449   }
00450   //[sec]
00451   m_interval = interval;
00452   //[Hz]
00453   m_dFreq = 1.0 / m_fftlen / interval;
