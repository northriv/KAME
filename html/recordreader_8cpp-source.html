00126                 + sizeof(int32_t) //time().sec()
00127                 + sizeof(int32_t) //time().usec()

00129                 + strlen(sup) //reserved
00130                 + 2 //two null chars
00131                 + sizeof(uint32_t)  //allsize
00132                 );
00133     if(driver) {

00142                  throw XIOError(__FILE__, __LINE__);
00143         throw XNoDriverError(name, __FILE__, __LINE__);
00144     }
00145     try {
00146         driver->clearRaw();

00148         if(gzread(fd, &driver->rawData()[0], size) == -1)
00149             throw XIOError(__FILE__, __LINE__);
