00016 double rootsDKA(const std::vector<std::complex<double> > &polynomial, std::vector<std::complex<double> > &roots,
00017         double eps, int max_it) {
00018         int n = polynomial.size() - 1;
00019         ASSERT(abs(polynomial[n] - 1.0) < 0.01);
00020         ASSERT(polynomial.size() == roots.size() + 1);
00021         std::vector<bool> conv(n, false);
00022         std::vector<std::complex<double> > roots_next(n);
00023         double err;     
00024         int it;
00025         for(it = 0; it < max_it; it++) {
00026                 err = 0.0;      
00027                 std::copy(roots.begin(), roots.end(), roots_next.begin());
00028                 for(int i = 0; i < n; i++) {
00029                         if(conv[i])
00030                                 continue;
00031                         std::complex<double> f = 0.0;
00032                         std::complex<double> x = roots[i];
00033                         std::complex<double> xn = 1.0;
00034                         for(int j = 0; j <= n; j++) {
00035                                 f += polynomial[j] * xn;
00036                                 xn *= x;
00037                         }
00038                         std::complex<double> deno = 1.0;
00039                         for(int j = 0; j < n; j++) {
00040                                 if(i == j) continue;
00041                                 deno *= (x - roots[j]);
00042                         }
00043                         if(std::abs(deno) < eps) {
00044                                 conv[i] = true;
00045                                 continue;
00046                         }
00047                         std::complex<double> dx = - f / deno;
00048                         err = std::max(std::abs(dx), err);
00049                         roots_next[i] = roots[i] + dx;
00050                 }
00051                 std::copy(roots_next.begin(), roots_next.end(), roots.begin());
00052                 if(err < eps)
00053                         break;
00054         }
00055         dbgPrint(formatString("Roots-DKA: it=%d, err=%g", it, err));
00056         return err;
00057 }
